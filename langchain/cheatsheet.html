LangChain Cheat Sheet

Quick reference for the most important LangChain patterns, classes, and APIs — all in one page.

Installation

pip install langchain langchain-openai langchain-anthropic langchain-community
pip install langchain-chroma faiss-cpu python-dotenv tiktoken

Models

from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from langchain_anthropic import ChatAnthropic
from langchain_community.chat_models import ChatOllama

# Chat models
openai_llm    = ChatOpenAI(model="gpt-4o-mini", temperature=0)
anthropic_llm = ChatAnthropic(model="claude-haiku-4-5", temperature=0)
local_llm     = ChatOllama(model="llama3.2", temperature=0)

# Embeddings
embeddings = OpenAIEmbeddings(model="text-embedding-3-small")

# Invoke
resp = openai_llm.invoke("Hello!")        # → AIMessage
resp = openai_llm.invoke([                 # or with messages
    ("system", "You are helpful."),
    ("human", "Hello!")
])
print(resp.content)

# Stream
for chunk in openai_llm.stream("Tell me a joke"):
    print(chunk.content, end="")

# Batch
results = openai_llm.batch(["Q1", "Q2", "Q3"])

Prompts

from langchain_core.prompts import (
    PromptTemplate, ChatPromptTemplate,
    MessagesPlaceholder, FewShotChatMessagePromptTemplate
)

# String prompt
p = PromptTemplate.from_template("Explain {topic} simply.")
p.invoke({"topic": "Docker"})

# Chat prompt
cp = ChatPromptTemplate.from_messages([
    ("system", "You are a {role}."),
    MessagesPlaceholder("history"),
    ("human", "{input}"),
])
cp.invoke({"role": "tutor", "history": [], "input": "What is LCEL?"})

# Partial prompt
dated = p.partial(topic="Kubernetes")
dated.invoke({})   # uses pre-filled topic

Output Parsers

from langchain_core.output_parsers import StrOutputParser, JsonOutputParser
from pydantic import BaseModel, Field

# String
chain = prompt | llm | StrOutputParser()     # returns str

# JSON
chain = prompt | llm | JsonOutputParser()    # returns dict

# Structured output (Pydantic)
class Answer(BaseModel):
    answer: str = Field(description="The answer")
    confidence: float = Field(description="0.0 to 1.0")

structured = llm.with_structured_output(Answer)
result = structured.invoke("What is Docker?")
print(result.answer, result.confidence)

LCEL Chains

from langchain_core.runnables import (
    RunnableParallel, RunnablePassthrough, RunnableLambda, RunnableBranch
)

# Sequential
chain = prompt | llm | StrOutputParser()
chain.invoke({"input": "..."})

# Parallel — runs simultaneously
parallel = RunnableParallel(
    a=chain_a,
    b=chain_b,
)
result = parallel.invoke({"input": "..."})
# {"a": ..., "b": ...}

# Passthrough — forward inputs unchanged
chain = RunnableParallel(
    original=RunnablePassthrough(),
    processed=some_chain,
)

# Lambda — wrap any function
f = RunnableLambda(lambda x: x["text"].upper())

# Branch — conditional routing
branch = RunnableBranch(
    (lambda x: len(x["text"]) > 100, long_chain),
    short_chain,   # default
)

# Fallback
resilient = primary_chain.with_fallbacks([fallback_chain])

# Retry
retry_chain = chain.with_retry(stop_after_attempt=3)

Memory & Chat History

from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.runnables.history import RunnableWithMessageHistory

# Session store
store = {}
def get_history(session_id):
    if session_id not in store:
        store[session_id] = InMemoryChatMessageHistory()
    return store[session_id]

# Wrap chain
chat_chain = RunnableWithMessageHistory(
    prompt | llm | StrOutputParser(),
    get_history,
    input_messages_key="input",
    history_messages_key="history",
)

chat_chain.invoke(
    {"input": "Hello!"},
    config={"configurable": {"session_id": "user_1"}},
)

RAG

from langchain_community.document_loaders import PyPDFLoader, WebBaseLoader
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_openai import OpenAIEmbeddings
from langchain_chroma import Chroma
from langchain_core.runnables import RunnablePassthrough

# 1. Load
docs = PyPDFLoader("file.pdf").load()

# 2. Split
chunks = RecursiveCharacterTextSplitter(
    chunk_size=1000, chunk_overlap=200
).split_documents(docs)

# 3. Store
vectorstore = Chroma.from_documents(
    chunks, OpenAIEmbeddings(), persist_directory="./db"
)

# 4. Retrieve
retriever = vectorstore.as_retriever(search_kwargs={"k": 4})

# 5. Generate
rag_chain = (
    {"context": retriever | (lambda docs: "\n\n".join(d.page_content for d in docs)),
     "question": RunnablePassthrough()}
    | ChatPromptTemplate.from_messages([
        ("system", "Answer using this context:\n{context}"),
        ("human", "{question}"),
    ])
    | llm | StrOutputParser()
)

answer = rag_chain.invoke("What does the document say about X?")

Agents & Tools

from langchain_core.tools import tool
from langchain.agents import create_tool_calling_agent, AgentExecutor
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder

# 1. Define tools
@tool
def search(query: str) -> str:
    """Search the web for information."""
    return f"Result for: {query}"

@tool
def calculate(expr: str) -> str:
    """Evaluate a math expression."""
    return str(eval(expr, {"__builtins__": {}}, {}))

# 2. Agent prompt (must have agent_scratchpad)
prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant."),
    ("human", "{input}"),
    MessagesPlaceholder("agent_scratchpad"),
])

# 3. Create and run
tools = [search, calculate]
agent = create_tool_calling_agent(llm, tools, prompt)
executor = AgentExecutor(agent=agent, tools=tools, verbose=True, max_iterations=10)
result = executor.invoke({"input": "What is 15 * 7?"})
