Prompt Templates

Prompt templates separate your prompt structure from dynamic input values. Instead of building strings by hand, you define a template once and fill it with variables at runtime — making prompts reusable, testable, and composable in chains.

PromptTemplate — simple string prompts

from langchain_core.prompts import PromptTemplate

# Define template with {variable} placeholders
template = PromptTemplate.from_template(
    "You are an expert in {topic}. Explain {concept} in simple terms."
)

# Fill the template
prompt = template.invoke({"topic": "Docker", "concept": "container networking"})
print(prompt.text)
# "You are an expert in Docker. Explain container networking in simple terms."

# Use in a chain
from langchain_openai import ChatOpenAI
llm = ChatOpenAI(model="gpt-4o-mini")
chain = template | llm
response = chain.invoke({"topic": "Kubernetes", "concept": "pod scheduling"})

ChatPromptTemplate — for chat models

from langchain_core.prompts import ChatPromptTemplate

# Define system + human messages
prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a {role}. Always respond in {language}."),
    ("human", "{question}"),
])

# Invoke with variables
messages = prompt.invoke({
    "role": "senior DevOps engineer",
    "language": "English",
    "question": "What is the difference between a Deployment and a StatefulSet?",
})

chain = prompt | llm
response = chain.invoke({
    "role": "Python tutor",
    "language": "English",
    "question": "What is a list comprehension?",
})

MessagesPlaceholder — inject conversation history

from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder

prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant."),
    MessagesPlaceholder("history"),   # ← insert chat history here
    ("human", "{input}"),
])

# history is a list of BaseMessage objects (AIMessage, HumanMessage)
chain = prompt | llm
response = chain.invoke({
    "history": [
        ("human", "My name is Alice."),
        ("assistant", "Nice to meet you, Alice!"),
    ],
    "input": "What's my name?",
})
# "Your name is Alice."

Output parsers — structured responses

from langchain_core.output_parsers import StrOutputParser, JsonOutputParser
from langchain_core.pydantic_v1 import BaseModel, Field
from typing import List

# 1. String parser — extract .content automatically
chain = prompt | llm | StrOutputParser()
result = chain.invoke({"question": "What is Python?"})
print(type(result))   # str (not AIMessage)

# 2. Pydantic structured output
class CodeReview(BaseModel):
    score: int = Field(description="Quality score 1-10")
    issues: List[str] = Field(description="List of issues found")
    suggestion: str = Field(description="Main improvement suggestion")

structured_llm = llm.with_structured_output(CodeReview)
review_prompt = ChatPromptTemplate.from_template(
    "Review this Python code and return structured feedback:\n\n{code}"
)
chain = review_prompt | structured_llm
result = chain.invoke({"code": "def add(a,b): return a+b"})
print(result.score)       # 7
print(result.issues)      # ["Missing type hints", "No docstring"]

Few-shot prompting

from langchain_core.prompts import FewShotChatMessagePromptTemplate

examples = [
    {"input": "2 + 2",    "output": "4"},
    {"input": "5 * 3",    "output": "15"},
    {"input": "10 / 2",   "output": "5"},
]

example_prompt = ChatPromptTemplate.from_messages([
    ("human", "{input}"),
    ("ai",    "{output}"),
])

few_shot_prompt = FewShotChatMessagePromptTemplate(
    example_prompt=example_prompt,
    examples=examples,
)

final_prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a calculator. Only return the number."),
    few_shot_prompt,
    ("human", "{input}"),
])

chain = final_prompt | llm | StrOutputParser()
print(chain.invoke({"input": "8 + 7"}))   # "15"

Partial prompts — pre-fill some variables

from langchain_core.prompts import PromptTemplate
from datetime import datetime

# Pre-fill the date variable; {question} is filled at call time
prompt = PromptTemplate.from_template(
    "Today is {date}. Answer this question: {question}"
)

prompt_with_date = prompt.partial(date=datetime.now().strftime("%Y-%m-%d"))

chain = prompt_with_date | llm | StrOutputParser()
print(chain.invoke({"question": "What day of the week is it?"}))
Best practice: Always use ChatPromptTemplate with chat models — never build raw strings. It gives you cleaner role separation (system vs human vs assistant), easier few-shot examples, and seamless integration with memory and history.
