Memory

LLMs are stateless — each call forgets the previous one. Memory lets your chatbot or agent remember what was said, either by appending raw messages to the context window or by summarising them to fit within token limits.

The modern approach: ChatMessageHistory + LCEL

The recommended pattern is to manage chat history yourself using ChatMessageHistory and inject it via MessagesPlaceholder.

from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.runnables.history import RunnableWithMessageHistory

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful assistant."),
    MessagesPlaceholder("history"),
    ("human", "{input}"),
])

chain = prompt | llm

# Store for each session
store = {}
def get_session_history(session_id: str) -> InMemoryChatMessageHistory:
    if session_id not in store:
        store[session_id] = InMemoryChatMessageHistory()
    return store[session_id]

# Wrap the chain with history management
chat = RunnableWithMessageHistory(
    chain,
    get_session_history,
    input_messages_key="input",
    history_messages_key="history",
)

# Session "alice" — remembers across turns
config = {"configurable": {"session_id": "alice"}}

r1 = chat.invoke({"input": "My name is Alice."}, config=config)
print(r1.content)    # "Nice to meet you, Alice!"

r2 = chat.invoke({"input": "What is my name?"}, config=config)
print(r2.content)    # "Your name is Alice."

Buffer Window Memory — keep last N turns

from langchain_core.chat_history import InMemoryChatMessageHistory

class WindowedChatHistory(InMemoryChatMessageHistory):
    """Keeps only the last k messages to control token usage."""
    def __init__(self, k: int = 10):
        super().__init__()
        self.k = k

    def add_message(self, message):
        super().add_message(message)
        # Trim to last k messages
        if len(self.messages) > self.k:
            self.messages = self.messages[-self.k:]

store = {}
def get_windowed_history(session_id: str):
    if session_id not in store:
        store[session_id] = WindowedChatHistory(k=6)  # last 3 turns
    return store[session_id]

Persistent memory — Redis or database

from langchain_community.chat_message_histories import RedisChatMessageHistory

def get_redis_history(session_id: str):
    return RedisChatMessageHistory(
        session_id=session_id,
        url="redis://localhost:6379",
        ttl=3600,   # expire after 1 hour
    )

chat = RunnableWithMessageHistory(
    chain,
    get_redis_history,
    input_messages_key="input",
    history_messages_key="history",
)

Summarisation memory

from langchain.memory import ConversationSummaryBufferMemory

# Keeps raw messages until token limit, then summarises old ones
memory = ConversationSummaryBufferMemory(
    llm=llm,
    max_token_limit=500,
    return_messages=True,
    memory_key="history",
)

# Manually save and load messages
memory.save_context(
    {"input": "I'm building a RAG system with LangChain."},
    {"output": "Great! RAG combines retrieval with generation for accurate answers."},
)

saved = memory.load_memory_variables({})
print(saved["history"])   # list of messages (some may be a summary)

Multi-session chatbot (complete example)

from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.runnables.history import RunnableWithMessageHistory
from langchain_core.output_parsers import StrOutputParser

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.5)
prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a concise DevOps assistant. Keep answers to 2-3 sentences."),
    MessagesPlaceholder("history"),
    ("human", "{input}"),
])

store = {}
def get_history(sid):
    if sid not in store: store[sid] = InMemoryChatMessageHistory()
    return store[sid]

bot = RunnableWithMessageHistory(
    prompt | llm | StrOutputParser(),
    get_history, input_messages_key="input", history_messages_key="history",
)

def chat(session_id: str, user_input: str) -> str:
    return bot.invoke(
        {"input": user_input},
        config={"configurable": {"session_id": session_id}},
    )

# Two independent sessions
print(chat("user_1", "What is a Kubernetes Pod?"))
print(chat("user_2", "How does Docker networking work?"))
print(chat("user_1", "Can a Pod have multiple containers?"))   # remembers context
Token budget: Every message in history uses tokens. For GPT-4o at ~$5/1M tokens, a 100-turn conversation can cost 50 000+ tokens. Use windowed history or summarisation for long-running chat applications.
