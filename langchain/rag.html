RAG & Vector Stores

Retrieval-Augmented Generation (RAG) lets an LLM answer questions about your own data — documents, PDFs, databases — without retraining. You index your data into a vector store, retrieve the most relevant chunks at query time, and pass them to the LLM as context.

RAG pipeline overview

1. Load

  • PDF, Web, CSV, DB
  • Document loaders

2. Split

  • Chunk into segments
  • RecursiveCharacterTextSplitter

3. Embed

  • Convert to vectors
  • OpenAIEmbeddings

4. Store

  • Save to vector DB
  • Chroma / FAISS / Pinecone

5. Retrieve

  • Similarity search
  • Top-k relevant chunks

6. Generate

  • LLM + context + question
  • Grounded answer

Document loaders

from langchain_community.document_loaders import (
    PyPDFLoader,
    WebBaseLoader,
    CSVLoader,
    TextLoader,
    DirectoryLoader,
)

# Load a PDF
pdf_loader = PyPDFLoader("docs/kubernetes-guide.pdf")
pdf_docs = pdf_loader.load()   # list of Document objects

# Load a web page
web_loader = WebBaseLoader("https://docs.docker.com/get-started/")
web_docs = web_loader.load()

# Load a CSV
csv_loader = CSVLoader("data/products.csv", source_column="name")
csv_docs = csv_loader.load()

# Load all .md files in a folder
dir_loader = DirectoryLoader("docs/", glob="**/*.md", loader_cls=TextLoader)
all_docs = dir_loader.load()

print(f"Loaded {len(all_docs)} documents")
print(all_docs[0].page_content[:200])
print(all_docs[0].metadata)   # {"source": "docs/intro.md", "row": 0, ...}

Text splitters

from langchain_text_splitters import RecursiveCharacterTextSplitter, MarkdownHeaderTextSplitter

# Most versatile splitter — tries to split on paragraphs, then sentences, then chars
splitter = RecursiveCharacterTextSplitter(
    chunk_size=1000,       # target chunk size in characters
    chunk_overlap=200,     # overlap to preserve context at boundaries
    length_function=len,
)

chunks = splitter.split_documents(all_docs)
print(f"Split into {len(chunks)} chunks")

# For Markdown — split at headers to preserve structure
md_splitter = MarkdownHeaderTextSplitter(
    headers_to_split_on=[("##", "section"), ("###", "subsection")]
)
md_chunks = md_splitter.split_text(markdown_text)

Vector stores — embed and index

from langchain_openai import OpenAIEmbeddings
from langchain_chroma import Chroma

embeddings = OpenAIEmbeddings(model="text-embedding-3-small")

# Build vector store from documents (indexes all chunks)
vectorstore = Chroma.from_documents(
    documents=chunks,
    embedding=embeddings,
    persist_directory="./chroma_db",  # save to disk
    collection_name="devdocs",
)

# Similarity search
results = vectorstore.similarity_search("How do I create a Docker volume?", k=4)
for doc in results:
    print(doc.page_content[:200])
    print("---")

# Similarity search with relevance scores (0-1, higher = more similar)
results_with_scores = vectorstore.similarity_search_with_relevance_scores(
    "What is a Kubernetes Pod?", k=3
)

FAISS — in-memory vector search

from langchain_community.vectorstores import FAISS

# Build FAISS index (in-memory, no server needed)
faiss_store = FAISS.from_documents(chunks, embeddings)

# Save and load
faiss_store.save_local("faiss_index")
faiss_store = FAISS.load_local("faiss_index", embeddings, allow_dangerous_deserialization=True)

Complete RAG chain (LCEL)

from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from langchain_chroma import Chroma
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough

# Load existing vector store
vectorstore = Chroma(
    persist_directory="./chroma_db",
    embedding_function=OpenAIEmbeddings(model="text-embedding-3-small"),
    collection_name="devdocs",
)

retriever = vectorstore.as_retriever(search_kwargs={"k": 4})

# Prompt that includes retrieved context
rag_prompt = ChatPromptTemplate.from_messages([
    ("system", """You are a helpful assistant. Answer using ONLY the context below.
If the answer is not in the context, say "I don't know."

Context:
{context}"""),
    ("human", "{question}"),
])

def format_docs(docs):
    return "\n\n".join(doc.page_content for doc in docs)

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

# Full RAG chain
rag_chain = (
    {"context": retriever | format_docs, "question": RunnablePassthrough()}
    | rag_prompt
    | llm
    | StrOutputParser()
)

answer = rag_chain.invoke("How do Docker volumes work?")
print(answer)

Advanced retrievers

from langchain.retrievers import ContextualCompressionRetriever, MultiQueryRetriever
from langchain.retrievers.document_compressors import LLMChainExtractor

# 1. MultiQuery — generates multiple phrasings of the question to improve recall
multi_retriever = MultiQueryRetriever.from_llm(
    retriever=vectorstore.as_retriever(),
    llm=llm,
)

# 2. Contextual Compression — removes irrelevant parts of retrieved chunks
compressor = LLMChainExtractor.from_llm(llm)
compression_retriever = ContextualCompressionRetriever(
    base_compressor=compressor,
    base_retriever=vectorstore.as_retriever(search_kwargs={"k": 6}),
)

# 3. Max Marginal Relevance — balances relevance and diversity
mmr_retriever = vectorstore.as_retriever(
    search_type="mmr",
    search_kwargs={"k": 4, "fetch_k": 20, "lambda_mult": 0.7},
)
Chunk size matters: Too small (under 200 chars) loses context; too large (over 2000 chars) reduces precision. Start with 1000 chars + 200 overlap. For code files, use Language splitter so chunks stay on function/class boundaries.
