Testing

Spring Boot's test support makes writing unit and integration tests straightforward. The spring-boot-starter-test dependency bundles JUnit 5, Mockito, AssertJ, and MockMvc, covering everything from isolated unit tests to full server integration tests.

Test slice annotations

AnnotationWhat it loadsUse when
@SpringBootTestFull application contextIntegration tests — all beans available
@WebMvcTestMVC layer only (controllers, filters, advice)Testing controller logic without DB
@DataJpaTestJPA + in-memory H2 by defaultRepository query tests
@JsonTestJackson serialization / deserializationTesting JSON mapping
@RestClientTestRestTemplate / RestClient mock serverTesting HTTP client calls

Unit test — service layer

import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;

import static org.assertj.core.api.Assertions.*;
import static org.mockito.Mockito.*;

@ExtendWith(MockitoExtension.class)   // no Spring context — fast
class ProductServiceTest {

    @Mock
    ProductRepository repo;

    @InjectMocks
    ProductService service;

    @Test
    void findById_returnsProduct_whenFound() {
        var product = new Product(1L, "Widget", BigDecimal.valueOf(9.99));
        when(repo.findById(1L)).thenReturn(Optional.of(product));

        Optional<Product> result = service.findById(1L);

        assertThat(result).isPresent()
                          .get()
                          .extracting(Product::getName)
                          .isEqualTo("Widget");
        verify(repo).findById(1L);
    }

    @Test
    void save_throwsException_whenNameExists() {
        when(repo.existsByName("Widget")).thenReturn(true);

        assertThatThrownBy(() -> service.create(new CreateProductRequest("Widget", 9.99)))
                .isInstanceOf(DuplicateNameException.class)
                .hasMessageContaining("Widget");
    }
}

Controller test — @WebMvcTest + MockMvc

@WebMvcTest(ProductController.class)   // only loads MVC slice
class ProductControllerTest {

    @Autowired
    MockMvc mockMvc;

    @MockBean               // replaces the real ProductService with a Mockito mock
    ProductService service;

    @Autowired
    ObjectMapper objectMapper;

    @Test
    void getAll_returns200WithProductList() throws Exception {
        var products = List.of(new Product(1L, "Widget", BigDecimal.TEN));
        when(service.findAll()).thenReturn(products);

        mockMvc.perform(get("/api/products")
                    .contentType(MediaType.APPLICATION_JSON))
               .andExpect(status().isOk())
               .andExpect(jsonPath("$", hasSize(1)))
               .andExpect(jsonPath("$[0].name").value("Widget"))
               .andExpect(jsonPath("$[0].price").value(10));
    }

    @Test
    void create_returns201_withValidBody() throws Exception {
        var req = new CreateProductRequest("Gadget", 24.99);
        var saved = new Product(2L, "Gadget", BigDecimal.valueOf(24.99));
        when(service.save(any())).thenReturn(saved);

        mockMvc.perform(post("/api/products")
                    .contentType(MediaType.APPLICATION_JSON)
                    .content(objectMapper.writeValueAsString(req)))
               .andExpect(status().isCreated())
               .andExpect(jsonPath("$.id").value(2));
    }

    @Test
    void create_returns400_withMissingName() throws Exception {
        mockMvc.perform(post("/api/products")
                    .contentType(MediaType.APPLICATION_JSON)
                    .content("{\"price\": 9.99}"))
               .andExpect(status().isBadRequest())
               .andExpect(jsonPath("$.name").exists());
    }
}

Repository test — @DataJpaTest

@DataJpaTest   // uses in-memory H2 by default; rolls back after each test
class ProductRepositoryTest {

    @Autowired
    TestEntityManager em;

    @Autowired
    ProductRepository repo;

    @Test
    void findByName_returnsMatchingProducts() {
        em.persistAndFlush(new Product(null, "Widget", BigDecimal.TEN));
        em.persistAndFlush(new Product(null, "Gadget", BigDecimal.ONE));

        List<Product> result = repo.findByName("Widget");

        assertThat(result).hasSize(1)
                          .first()
                          .extracting(Product::getName)
                          .isEqualTo("Widget");
    }

    @Test
    void existsByName_returnsTrue_whenExists() {
        em.persistAndFlush(new Product(null, "Widget", BigDecimal.TEN));
        assertThat(repo.existsByName("Widget")).isTrue();
        assertThat(repo.existsByName("Unknown")).isFalse();
    }
}

Integration test — @SpringBootTest

@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
class ProductIntegrationTest {

    @Autowired
    TestRestTemplate restTemplate;

    @Autowired
    ProductRepository repo;

    @BeforeEach
    void setUp() {
        repo.deleteAll();
        repo.save(new Product(null, "Widget", BigDecimal.TEN));
    }

    @Test
    void getAll_returnsAllProducts() {
        ResponseEntity<List<Product>> response = restTemplate
                .exchange("/api/products", HttpMethod.GET, null,
                          new ParameterizedTypeReference<>() {});

        assertThat(response.getStatusCode()).isEqualTo(HttpStatus.OK);
        assertThat(response.getBody()).hasSize(1);
    }
}

Testcontainers — real database in tests

@SpringBootTest
@Testcontainers
class ProductRepositoryIT {

    @Container
    static PostgreSQLContainer<?> postgres =
            new PostgreSQLContainer<>("postgres:16-alpine");

    @DynamicPropertySource
    static void configure(DynamicPropertyRegistry registry) {
        registry.add("spring.datasource.url", postgres::getJdbcUrl);
        registry.add("spring.datasource.username", postgres::getUsername);
        registry.add("spring.datasource.password", postgres::getPassword);
    }

    @Autowired ProductRepository repo;

    @Test
    void persistAndRetrieve() {
        var saved = repo.save(new Product(null, "Widget", BigDecimal.TEN));
        assertThat(repo.findById(saved.getId())).isPresent();
    }
}
Testing pyramid: write many fast unit tests (no Spring context), fewer @WebMvcTest / @DataJpaTest slice tests, and a handful of full @SpringBootTest integration tests. Testcontainers moves the DB integration tests closer to reality without external infrastructure.
