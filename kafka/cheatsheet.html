Kafka CLI Cheat Sheet

All the shell commands you'll reach for daily — topic management, producing, consuming, consumer group inspection, offset resets, and common configs.

Shorthand: replace kafka-topics.sh with kafka-topics (no extension) on installations that include the shell wrapper. All commands assume --bootstrap-server localhost:9092; set an alias: alias kbs="--bootstrap-server localhost:9092".

Topics

# List all topics
kafka-topics.sh --bootstrap-server localhost:9092 --list

# Create
kafka-topics.sh --bootstrap-server localhost:9092 \
  --create --topic my-topic --partitions 6 --replication-factor 3

# Describe (shows partition/replica/ISR layout)
kafka-topics.sh --bootstrap-server localhost:9092 \
  --describe --topic my-topic

# Describe all topics
kafka-topics.sh --bootstrap-server localhost:9092 --describe

# Alter partition count (can only increase)
kafka-topics.sh --bootstrap-server localhost:9092 \
  --alter --topic my-topic --partitions 12

# Alter topic config
kafka-configs.sh --bootstrap-server localhost:9092 \
  --alter --entity-type topics --entity-name my-topic \
  --add-config retention.ms=86400000

# Delete (requires delete.topic.enable=true, which is default)
kafka-topics.sh --bootstrap-server localhost:9092 \
  --delete --topic my-topic

Produce & consume (console tools)

# Produce — type messages, press Enter for each
kafka-console-producer.sh --bootstrap-server localhost:9092 \
  --topic my-topic

# Produce with key (key:value separated by ":")
kafka-console-producer.sh --bootstrap-server localhost:9092 \
  --topic my-topic \
  --property parse.key=true \
  --property key.separator=:

# Consume from beginning
kafka-console-consumer.sh --bootstrap-server localhost:9092 \
  --topic my-topic --from-beginning

# Consume and print key, value, timestamp, offset
kafka-console-consumer.sh --bootstrap-server localhost:9092 \
  --topic my-topic --from-beginning \
  --property print.key=true \
  --property print.timestamp=true \
  --property print.offset=true \
  --property key.separator=" | "

# Consume with a consumer group
kafka-console-consumer.sh --bootstrap-server localhost:9092 \
  --topic my-topic --group my-group

# Consume only N messages then exit
kafka-console-consumer.sh --bootstrap-server localhost:9092 \
  --topic my-topic --from-beginning --max-messages 100

Consumer groups

# List all consumer groups
kafka-consumer-groups.sh --bootstrap-server localhost:9092 --list

# Describe a group (shows lag per partition)
kafka-consumer-groups.sh --bootstrap-server localhost:9092 \
  --group my-group --describe

# Describe all groups
kafka-consumer-groups.sh --bootstrap-server localhost:9092 \
  --all-groups --describe

# Delete a consumer group (must be inactive)
kafka-consumer-groups.sh --bootstrap-server localhost:9092 \
  --group my-group --delete

Offset management

# Reset offsets to earliest (dry run first)
kafka-consumer-groups.sh --bootstrap-server localhost:9092 \
  --group my-group --topic my-topic \
  --reset-offsets --to-earliest --dry-run

# Apply the reset
kafka-consumer-groups.sh --bootstrap-server localhost:9092 \
  --group my-group --topic my-topic \
  --reset-offsets --to-earliest --execute

# Reset to specific offset
kafka-consumer-groups.sh --bootstrap-server localhost:9092 \
  --group my-group --topic my-topic:0 \
  --reset-offsets --to-offset 1000 --execute

# Reset to a datetime
kafka-consumer-groups.sh --bootstrap-server localhost:9092 \
  --group my-group --topic my-topic \
  --reset-offsets --to-datetime 2024-01-01T00:00:00.000 --execute

# Reset by shifting N messages back
kafka-consumer-groups.sh --bootstrap-server localhost:9092 \
  --group my-group --topic my-topic \
  --reset-offsets --shift-by -500 --execute

Cluster and broker info

# Describe cluster (lists all brokers)
kafka-broker-api-versions.sh --bootstrap-server localhost:9092

# Show broker configs
kafka-configs.sh --bootstrap-server localhost:9092 \
  --describe --entity-type brokers --entity-name 1

# Show topic configs (non-default)
kafka-configs.sh --bootstrap-server localhost:9092 \
  --describe --entity-type topics --entity-name my-topic

# Check leader election
kafka-leader-election.sh --bootstrap-server localhost:9092 \
  --election-type preferred --all-topic-partitions

Log inspection

# Print earliest/latest offsets per partition
kafka-run-class.sh kafka.tools.GetOffsetShell \
  --bootstrap-server localhost:9092 --topic my-topic

# Dump segment file contents (raw)
kafka-dump-log.sh \
  --files /var/kafka-logs/my-topic-0/00000000000000000000.log \
  --print-data-log

# Dump with deep iteration (decode values)
kafka-dump-log.sh \
  --files /var/kafka-logs/my-topic-0/00000000000000000000.log \
  --deep-iteration --print-data-log

Producer / consumer performance tests

# Produce 1M 100-byte messages as fast as possible
kafka-producer-perf-test.sh \
  --topic perf-test --num-records 1000000 \
  --record-size 100 --throughput -1 \
  --producer-props bootstrap.servers=localhost:9092

# Consume and measure throughput
kafka-consumer-perf-test.sh \
  --bootstrap-server localhost:9092 \
  --topic perf-test --messages 1000000
