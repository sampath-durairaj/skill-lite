RAG Cheat Sheet

Complete reference for all RAG patterns, APIs, and configurations on one page.

Full indexing pipeline

from langchain_community.document_loaders import PyPDFLoader, DirectoryLoader, WebBaseLoader
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_openai import OpenAIEmbeddings
from langchain_chroma import Chroma

# 1. Load
docs = PyPDFLoader("file.pdf").load()
# or: DirectoryLoader("docs/", glob="**/*.md").load()
# or: WebBaseLoader(["https://example.com"]).load()

# 2. Split
chunks = RecursiveCharacterTextSplitter(
    chunk_size=1000, chunk_overlap=200
).split_documents(docs)

# 3. Embed + Store
db = Chroma.from_documents(
    documents=chunks,
    embedding=OpenAIEmbeddings(model="text-embedding-3-small"),
    persist_directory="./chroma_db",
    collection_name="my_docs",
)

Full RAG chain (LCEL)

from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough

# Load existing store
db = Chroma(persist_directory="./chroma_db", embedding_function=OpenAIEmbeddings())
retriever = db.as_retriever(search_kwargs={"k": 4})
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

prompt = ChatPromptTemplate.from_messages([
    ("system", """Answer using ONLY the context below.
If the answer is not in the context, say "I don't know."

Context:
{context}"""),
    ("human", "{question}"),
])

def format_docs(docs):
    return "\n\n".join(doc.page_content for doc in docs)

rag_chain = (
    {"context": retriever | format_docs, "question": RunnablePassthrough()}
    | prompt | llm | StrOutputParser()
)

answer = rag_chain.invoke("What is the vacation policy?")

Text splitters quick reference

from langchain_text_splitters import (
    RecursiveCharacterTextSplitter,    # general (default)
    MarkdownHeaderTextSplitter,        # Markdown files
    Language,                          # source code
)
from langchain_experimental.text_splitter import SemanticChunker  # semantic

# General
splitter = RecursiveCharacterTextSplitter(chunk_size=1000, chunk_overlap=200)

# Markdown
md = MarkdownHeaderTextSplitter([("#","h1"),("##","h2"),("###","h3")])

# Code
py = RecursiveCharacterTextSplitter.from_language(Language.PYTHON, chunk_size=500)

# Token-based
tok = RecursiveCharacterTextSplitter.from_tiktoken_encoder("gpt-4o", chunk_size=256)

Retrievers quick reference

from langchain.retrievers import MultiQueryRetriever, ContextualCompressionRetriever, EnsembleRetriever, ParentDocumentRetriever
from langchain.retrievers.document_compressors import LLMChainExtractor, EmbeddingsFilter
from langchain_community.retrievers import BM25Retriever

# Basic similarity
r = db.as_retriever(search_kwargs={"k": 4})

# MMR (diverse results)
r = db.as_retriever(search_type="mmr", search_kwargs={"k":4,"fetch_k":20})

# Score threshold
r = db.as_retriever(search_type="similarity_score_threshold",
                    search_kwargs={"k":4,"score_threshold":0.7})

# MultiQuery (LLM rewrites question)
r = MultiQueryRetriever.from_llm(retriever=r, llm=llm)

# Compression (LLM trims irrelevant content)
r = ContextualCompressionRetriever(
    base_compressor=LLMChainExtractor.from_llm(llm), base_retriever=r)

# Hybrid BM25 + vector
r = EnsembleRetriever(
    retrievers=[BM25Retriever.from_documents(chunks), db.as_retriever()],
    weights=[0.4, 0.6])
