Retrievers & Search

A retriever takes a query string and returns the most relevant documents. LangChain's Retriever interface is the glue between your vector store and your generation chain — and there are many strategies beyond basic similarity search that dramatically improve retrieval quality.

Basic vector store retriever

from langchain_chroma import Chroma
from langchain_openai import OpenAIEmbeddings

db = Chroma(persist_directory="./chroma_db", embedding_function=OpenAIEmbeddings())

# Convert vector store to a retriever
retriever = db.as_retriever(
    search_type="similarity",    # default
    search_kwargs={"k": 4},      # return 4 most similar chunks
)

# Use it directly
docs = retriever.invoke("How does Docker networking work?")
for doc in docs:
    print(f"[{doc.metadata.get('source')}] {doc.page_content[:100]}")

Search types

# 1. Similarity — pure cosine/L2 distance
retriever = db.as_retriever(
    search_type="similarity",
    search_kwargs={"k": 4},
)

# 2. Similarity with score threshold — filter out low-quality matches
retriever = db.as_retriever(
    search_type="similarity_score_threshold",
    search_kwargs={"k": 6, "score_threshold": 0.7},
)

# 3. MMR (Maximal Marginal Relevance) — balances relevance AND diversity
# Avoids returning 4 near-identical chunks
retriever = db.as_retriever(
    search_type="mmr",
    search_kwargs={
        "k": 4,           # return 4 docs
        "fetch_k": 20,    # consider 20 candidates
        "lambda_mult": 0.7,  # 1.0 = pure relevance, 0.0 = pure diversity
    },
)

MultiQueryRetriever — overcome query phrasing sensitivity

from langchain.retrievers import MultiQueryRetriever
from langchain_openai import ChatOpenAI

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

# Generates 3-5 alternative phrasings of the question,
# retrieves for each, and deduplicates results
retriever = MultiQueryRetriever.from_llm(
    retriever=db.as_retriever(search_kwargs={"k": 3}),
    llm=llm,
)

# For "Docker networking" it might also search:
# "How do containers communicate?" / "Docker bridge networks" / "container port mapping"
docs = retriever.invoke("Docker networking")

ContextualCompressionRetriever — remove irrelevant content

from langchain.retrievers import ContextualCompressionRetriever
from langchain.retrievers.document_compressors import (
    LLMChainExtractor, EmbeddingsFilter, DocumentCompressorPipeline
)
from langchain_openai import OpenAIEmbeddings

# Option 1: LLM extracts only relevant sentences from each chunk
compressor = LLMChainExtractor.from_llm(llm)
compression_retriever = ContextualCompressionRetriever(
    base_compressor=compressor,
    base_retriever=db.as_retriever(search_kwargs={"k": 6}),
)

# Option 2: Embedding filter — faster, no LLM cost
embeddings_filter = EmbeddingsFilter(
    embeddings=OpenAIEmbeddings(),
    similarity_threshold=0.76,
)
compression_retriever = ContextualCompressionRetriever(
    base_compressor=embeddings_filter,
    base_retriever=db.as_retriever(search_kwargs={"k": 6}),
)

Ensemble / hybrid retrieval (BM25 + vector)

pip install rank-bm25
from langchain.retrievers import EnsembleRetriever
from langchain_community.retrievers import BM25Retriever

# BM25 is keyword-based (exact match strength)
bm25_retriever   = BM25Retriever.from_documents(chunks)
bm25_retriever.k = 4

# Vector retriever (semantic match strength)
vector_retriever = db.as_retriever(search_kwargs={"k": 4})

# Combine: BM25 handles exact terms, vector handles synonyms/meaning
ensemble = EnsembleRetriever(
    retrievers=[bm25_retriever, vector_retriever],
    weights=[0.4, 0.6],   # 40% BM25, 60% vector
)

docs = ensemble.invoke("docker-compose networking port mapping")

Self-querying retriever — filter from natural language

from langchain.retrievers.self_query.base import SelfQueryRetriever
from langchain.chains.query_constructor.base import AttributeInfo

metadata_fields = [
    AttributeInfo(name="source",     description="Source document filename", type="string"),
    AttributeInfo(name="page",       description="Page number",              type="integer"),
    AttributeInfo(name="department", description="Owning department",        type="string"),
    AttributeInfo(name="year",       description="Publication year",         type="integer"),
]

retriever = SelfQueryRetriever.from_llm(
    llm=llm,
    vectorstore=db,
    document_contents="HR policies and company handbook",
    metadata_field_info=metadata_fields,
)

# LLM converts this into a structured filter automatically:
# → similarity search + filter(department="HR", year=2025)
docs = retriever.invoke("Show me HR vacation policies from 2025")
Retrieval strategy guide: Start with basic similarity + MMR. Add MultiQuery if users phrase questions inconsistently. Add BM25 ensemble if your documents contain exact product names, IDs, or codes. Add ContextualCompression when retrieved chunks have too much irrelevant padding. Layer these — they compose cleanly.
