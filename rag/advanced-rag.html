Advanced RAG

When basic RAG isn't accurate enough, advanced techniques improve retrieval precision, handle complex queries, and reduce hallucination. These include hypothetical document embeddings, re-ranking, parent document retrieval, CRAG, and Agentic RAG with LangGraph.

HyDE — Hypothetical Document Embeddings

Instead of embedding the user's short question, use the LLM to generate a hypothetical answer first, then embed that. A hypothetical answer is semantically closer to real document chunks than a short question.

from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from langchain_chroma import Chroma
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
db  = Chroma(persist_directory="./db", embedding_function=OpenAIEmbeddings())

# Step 1: generate a hypothetical document that would answer the query
hyde_prompt = ChatPromptTemplate.from_template(
    "Write a short passage that directly answers this question. Be specific:\n\n{question}"
)
hyde_chain = hyde_prompt | llm | StrOutputParser()

# Step 2: use the hypothetical answer as the retrieval query
def hyde_retrieve(question: str):
    hypothetical = hyde_chain.invoke({"question": question})
    return db.similarity_search(hypothetical, k=4)

# In a full RAG chain:
rag_prompt = ChatPromptTemplate.from_messages([
    ("system", "Answer using ONLY this context:\n{context}"),
    ("human", "{question}"),
])

rag_chain = (
    {"context": lambda x: "\n\n".join(d.page_content for d in hyde_retrieve(x)),
     "question": RunnablePassthrough()}
    | rag_prompt | llm | StrOutputParser()
)

print(rag_chain.invoke("What are the Docker networking modes?"))

Re-ranking with cross-encoders

A bi-encoder (embedding) retrieves fast but approximates relevance. A cross-encoder re-scores each (query, document) pair precisely — much slower but more accurate. Use cross-encoders on a small candidate set after initial retrieval.

from sentence_transformers import CrossEncoder
from langchain_chroma import Chroma
from langchain_openai import OpenAIEmbeddings

cross_encoder = CrossEncoder("cross-encoder/ms-marco-MiniLM-L-6-v2")
db = Chroma(persist_directory="./db", embedding_function=OpenAIEmbeddings())

def rerank_retrieve(query: str, k_initial: int = 20, k_final: int = 4):
    # Step 1: fast bi-encoder retrieval — get 20 candidates
    candidates = db.similarity_search(query, k=k_initial)

    # Step 2: score each (query, document) pair with cross-encoder
    pairs  = [(query, doc.page_content) for doc in candidates]
    scores = cross_encoder.predict(pairs)

    # Step 3: sort by cross-encoder score, keep top k_final
    ranked = sorted(zip(scores, candidates), key=lambda x: x[0], reverse=True)
    return [doc for _, doc in ranked[:k_final]]

docs = rerank_retrieve("How do Docker bridge networks work?")

Parent Document Retriever

Index small chunks for precise retrieval, but return the larger parent document for richer context.

from langchain.retrievers import ParentDocumentRetriever
from langchain.storage import InMemoryStore
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_chroma import Chroma
from langchain_openai import OpenAIEmbeddings

# Small chunks for precise retrieval
child_splitter  = RecursiveCharacterTextSplitter(chunk_size=200)
# Larger chunks returned to the LLM
parent_splitter = RecursiveCharacterTextSplitter(chunk_size=2000)

vectorstore   = Chroma(embedding_function=OpenAIEmbeddings(), collection_name="child_docs")
docstore      = InMemoryStore()   # stores parent docs

retriever = ParentDocumentRetriever(
    vectorstore=vectorstore,
    docstore=docstore,
    child_splitter=child_splitter,
    parent_splitter=parent_splitter,
)

# Index documents
from langchain_community.document_loaders import PyPDFLoader
docs = PyPDFLoader("manual.pdf").load()
retriever.add_documents(docs)

# Retrieves small chunks → returns their larger parents
results = retriever.invoke("Docker networking bridge mode")

CRAG — Corrective RAG

CRAG grades retrieved documents and falls back to web search if they're irrelevant.

from langgraph.graph import StateGraph, START, END
from typing import TypedDict, List, Annotated
from langgraph.graph.message import add_messages
from langchain_core.documents import Document

class CRAGState(TypedDict):
    question: str
    documents: List[Document]
    grade: str        # "relevant" | "irrelevant"
    answer: str

def retrieve(state: CRAGState) -> dict:
    docs = db.similarity_search(state["question"], k=4)
    return {"documents": docs}

def grade_documents(state: CRAGState) -> dict:
    grade_prompt = ChatPromptTemplate.from_template(
        """Are these documents relevant to answer: "{question}"?
        Documents: {docs}
        Reply with ONLY "relevant" or "irrelevant"."""
    )
    chain  = grade_prompt | llm | StrOutputParser()
    result = chain.invoke({
        "question": state["question"],
        "docs": "\n".join(d.page_content[:200] for d in state["documents"]),
    })
    return {"grade": result.strip().lower()}

def web_search(state: CRAGState) -> dict:
    from langchain_community.tools import DuckDuckGoSearchRun
    results = DuckDuckGoSearchRun().run(state["question"])
    return {"documents": [Document(page_content=results)]}

def generate(state: CRAGState) -> dict:
    context = "\n\n".join(d.page_content for d in state["documents"])
    answer  = llm.invoke(f"Answer using this context:\n{context}\n\nQuestion: {state['question']}")
    return {"answer": answer.content}

g = StateGraph(CRAGState)
g.add_node("retrieve",   retrieve)
g.add_node("grade",      grade_documents)
g.add_node("web_search", web_search)
g.add_node("generate",   generate)
g.add_edge(START, "retrieve")
g.add_edge("retrieve", "grade")
g.add_conditional_edges("grade",
    lambda s: s["grade"],
    {"relevant": "generate", "irrelevant": "web_search"},
)
g.add_edge("web_search", "generate")
g.add_edge("generate", END)
crag_app = g.compile()
result = crag_app.invoke({"question": "What is Docker Swarm?", "documents": [], "grade": "", "answer": ""})
When to use advanced techniques: Measure your baseline first with RAGAS or LangSmith. If precision is low → add re-ranking. If short queries miss long answers → try HyDE. If retrieved chunks are too small → use Parent Document Retriever. If your knowledge base is incomplete → CRAG with web search fallback. Don't add complexity until you can measure the improvement.
