Document Loaders

Document loaders ingest raw data from any source — PDFs, websites, databases, APIs, cloud storage — and convert it into LangChain Document objects: a page_content string plus a metadata dict. Every loader returns a list of Documents.

The Document object

from langchain_core.documents import Document

doc = Document(
    page_content="Docker containers are isolated environments...",
    metadata={
        "source": "docs/docker-guide.pdf",
        "page": 3,
        "author": "Jane Smith",
    }
)
print(doc.page_content)   # the text
print(doc.metadata)       # {"source": ..., "page": 3, ...}

PDF loaders

pip install pypdf          # PyPDFLoader (lightweight)
pip install pdfminer.six   # PDFMinerLoader (better layout)
pip install pymupdf        # PyMuPDFLoader (fastest, best quality)
from langchain_community.document_loaders import (
    PyPDFLoader, PDFMinerLoader, PyMuPDFLoader
)

# One Document per page
loader = PyPDFLoader("report.pdf")
pages  = loader.load()
print(f"Loaded {len(pages)} pages")
print(pages[0].metadata)  # {"source": "report.pdf", "page": 0}

# Better text extraction (handles tables, columns)
loader = PyMuPDFLoader("report.pdf")
pages  = loader.load()

# Lazy loading (memory-efficient for large files)
for page in loader.lazy_load():
    process(page)

Web loaders

pip install beautifulsoup4 requests   # WebBaseLoader
pip install playwright                # PlaywrightURLLoader (JavaScript sites)
from langchain_community.document_loaders import (
    WebBaseLoader, RecursiveUrlLoader, SitemapLoader
)
import bs4

# Load a single page (static HTML)
loader = WebBaseLoader(
    web_paths=["https://docs.langchain.com/docs/"],
    bs_kwargs={"parse_only": bs4.SoupStrainer(class_=("post-content", "post-title"))},
)
docs = loader.load()

# Crawl an entire site recursively (respects robots.txt)
loader = RecursiveUrlLoader(
    url="https://docs.docker.com/",
    max_depth=3,
    extractor=lambda x: BeautifulSoup(x, "html.parser").text,
)
docs = loader.load()

# Load from a sitemap
loader = SitemapLoader(web_path="https://docs.langchain.com/sitemap.xml")
docs = loader.load()

File format loaders

from langchain_community.document_loaders import (
    CSVLoader, JSONLoader, UnstructuredMarkdownLoader,
    UnstructuredWordDocumentLoader, UnstructuredExcelLoader,
    TextLoader,
)

# CSV — one Document per row
loader = CSVLoader("data.csv", source_column="id")
rows = loader.load()

# JSON — use jq to extract fields
loader = JSONLoader(
    file_path="data.json",
    jq_schema=".records[].content",
    text_content=False,
)
docs = loader.load()

# Markdown
loader = UnstructuredMarkdownLoader("README.md")
docs  = loader.load()

# Word / Excel (requires unstructured)
loader = UnstructuredWordDocumentLoader("report.docx")
docs  = loader.load()

# Plain text
loader = TextLoader("notes.txt", encoding="utf-8")
docs  = loader.load()

Directory loader — load entire folders

from langchain_community.document_loaders import DirectoryLoader, TextLoader, PyPDFLoader

# All .md files in a directory (recursive)
loader = DirectoryLoader(
    path="docs/",
    glob="**/*.md",
    loader_cls=TextLoader,
    loader_kwargs={"encoding": "utf-8"},
    show_progress=True,
    use_multithreading=True,   # parallel loading
)
docs = loader.load()
print(f"Loaded {len(docs)} documents")

# Mixed: PDFs in one folder
pdf_loader = DirectoryLoader("reports/", glob="**/*.pdf", loader_cls=PyPDFLoader)
pdfs = pdf_loader.load()

Database loaders

pip install sqlalchemy psycopg2-binary
from langchain_community.document_loaders import SQLDatabaseLoader
from langchain_community.utilities import SQLDatabase

db = SQLDatabase.from_uri("postgresql://user:pass@localhost/mydb")
loader = SQLDatabaseLoader(
    query="SELECT id, title, body FROM articles WHERE published = true",
    db=db,
    page_content_columns=["title", "body"],
    metadata_columns=["id"],
)
docs = loader.load()

Cloud storage loaders

from langchain_community.document_loaders import (
    S3FileLoader, S3DirectoryLoader,
    GCSFileLoader,
)

# AWS S3
loader = S3FileLoader(bucket="my-bucket", key="docs/manual.pdf")
docs = loader.load()

loader = S3DirectoryLoader(bucket="my-bucket", prefix="docs/")
docs = loader.load()

# Google Cloud Storage
loader = GCSFileLoader(project_name="my-project", bucket="my-bucket", blob="file.pdf")
docs = loader.load()

Adding custom metadata

from langchain_community.document_loaders import PyPDFLoader
from datetime import datetime

loader = PyPDFLoader("handbook.pdf")
docs   = loader.load()

# Enrich metadata after loading
for doc in docs:
    doc.metadata.update({
        "department": "HR",
        "indexed_at": datetime.now().isoformat(),
        "version": "2025-Q1",
    })
    # Optionally clean the text
    doc.page_content = doc.page_content.strip().replace("\x00", "")
Loader selection guide: Use PyMuPDFLoader for scanned PDFs (best OCR). Use WebBaseLoader for static sites, PlaywrightURLLoader for JavaScript-rendered pages. For large directories, always set use_multithreading=True — it cuts load time 4–8× on multi-core machines.
