LangGraph Cheat Sheet

Quick reference for all LangGraph patterns, classes, and APIs in one page.

Installation

pip install langgraph langchain-openai
pip install langgraph-checkpoint-sqlite   # persistent checkpointing

Graph skeleton

from typing import TypedDict, Annotated, List
from langgraph.graph import StateGraph, START, END, MessagesState
from langgraph.graph.message import add_messages
from langchain_core.messages import BaseMessage

# 1. State
class State(TypedDict):
    messages: Annotated[List[BaseMessage], add_messages]
    counter: int

# 2. Nodes
def my_node(state: State) -> dict:
    return {"counter": state["counter"] + 1}

# 3. Build
graph = StateGraph(State)
graph.add_node("my_node", my_node)
graph.add_edge(START, "my_node")
graph.add_edge("my_node", END)

# 4. Compile
app = graph.compile()

# 5. Run
result = app.invoke({"messages": [], "counter": 0})

State reducers

import operator
from langgraph.graph.message import add_messages

class State(TypedDict):
    messages:  Annotated[List[BaseMessage], add_messages]  # appends
    totals:    Annotated[int, operator.add]                 # sums
    items:     Annotated[List[str], operator.add]           # extends list
    best:      Annotated[float, max]                        # keeps max
    overwrite: str                                          # default: replaces

Edges

# Fixed edge
graph.add_edge("a", "b")
graph.add_edge(START, "first_node")
graph.add_edge("last_node", END)

# Conditional edge — router returns node name
def router(state: State) -> str:
    return "node_b" if state["counter"] > 5 else "node_c"

graph.add_conditional_edges("a", router)

# Conditional with explicit mapping
graph.add_conditional_edges(
    "a", router,
    {"node_b": "node_b", "node_c": "node_c", "done": END}
)

# Fan-out: run b and c in parallel after a
graph.add_edge("a", "b")
graph.add_edge("a", "c")

Checkpointing

from langgraph.checkpoint.memory import MemorySaver
from langgraph.checkpoint.sqlite import SqliteSaver

# Dev: in-memory
app = graph.compile(checkpointer=MemorySaver())

# Prod: SQLite
with SqliteSaver.from_conn_string("state.db") as cp:
    app = graph.compile(checkpointer=cp)

# Always pass thread_id
config = {"configurable": {"thread_id": "session_1"}}
app.invoke(input_state, config=config)

# Inspect state
snapshot = app.get_state(config)
snapshot.values   # full state
snapshot.next     # next nodes (empty if done)

# History / time travel
history = list(app.get_state_history(config))
app.invoke(None, config=history[2].config)   # replay from step 2

# Manual state update
app.update_state(config, {"counter": 0})

Human-in-the-loop

from langgraph.types import interrupt, Command

# Compile with breakpoint
app = graph.compile(
    checkpointer=MemorySaver(),
    interrupt_before=["send_email"],   # pause before this node
    # interrupt_after=["draft"],       # or pause after
)

# Run until interrupt
app.invoke(initial_state, config=config)

# Inspect, optionally update
app.update_state(config, new_partial_state)

# Resume
app.invoke(None, config=config)

# Dynamic interrupt inside a node
def approval_node(state):
    decision = interrupt({"message": "Approve this action?"})
    return {"approved": decision == "yes"}

# Resume with answer
app.invoke(Command(resume="yes"), config=config)

Streaming

# Full state after each node
for s in app.stream(input, stream_mode="values"):
    print(s)

# State delta per node
for update in app.stream(input, stream_mode="updates"):
    node, delta = list(update.items())[0]
    print(node, delta)

# LLM tokens
for chunk, meta in app.stream(input, stream_mode="messages"):
    print(chunk.content, end="")

# Async
async for chunk, meta in app.astream(input, stream_mode="messages"):
    print(chunk.content, end="")

# Rich events
async for event in app.astream_events(input, version="v2"):
    if event["event"] == "on_chat_model_stream":
        print(event["data"]["chunk"].content, end="")

Multi-agent

from langgraph.types import Command

# Handoff via Command
def agent_a(state):
    return Command(goto="agent_b", update={"messages": [...]})

# Supervisor routing
def supervisor(state):
    choice = llm.invoke(...)   # decides which agent
    return {"next": choice.content}

graph.add_conditional_edges("supervisor",
    lambda s: s["next"],
    {"agent_a": "agent_a", "agent_b": "agent_b", "FINISH": END}
)

# Subgraph as node
sub_app = sub_graph.compile()
parent_graph.add_node("sub", lambda s: sub_app.invoke({"q": s["task"]}))
