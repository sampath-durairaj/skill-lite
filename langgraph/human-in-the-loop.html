Human-in-the-Loop

Human-in-the-loop (HITL) lets you pause graph execution at a defined breakpoint, show the current state to a human, optionally modify it, and then resume — or abort. This is essential for high-stakes actions like sending emails, writing to databases, or making purchases.

How it works

1. Run graph

  • Execution starts normally
  • Graph runs until breakpoint

2. Pause

  • State saved to checkpointer
  • Returns to caller with __interrupt__

3. Human review

  • Inspect / modify state
  • Approve or reject

4. Resume

  • Call invoke(None, config)
  • Continues from breakpoint

interrupt_before — pause before a node runs

from langgraph.graph import StateGraph, START, END, MessagesState
from langgraph.checkpoint.memory import MemorySaver
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage

llm = ChatOpenAI(model="gpt-4o-mini")

def draft_email(state: MessagesState) -> dict:
    prompt = "Draft a professional email based on: " + state["messages"][-1].content
    draft = llm.invoke(prompt)
    return {"messages": [AIMessage(content=f"DRAFT:\n{draft.content}")]}

def send_email(state: MessagesState) -> dict:
    # In production: actually send the email
    print("Email sent!")
    return {"messages": [AIMessage(content="Email sent successfully.")]}

graph = StateGraph(MessagesState)
graph.add_node("draft", draft_email)
graph.add_node("send",  send_email)
graph.add_edge(START, "draft")
graph.add_edge("draft", "send")
graph.add_edge("send", END)

checkpointer = MemorySaver()
# interrupt_before="send" means: pause BEFORE the "send" node runs
app = graph.compile(checkpointer=checkpointer, interrupt_before=["send"])

config = {"configurable": {"thread_id": "email_task_1"}}

# Step 1: Run — graph will pause before "send"
result = app.invoke(
    {"messages": [HumanMessage("Write an email to the team about the new deployment.")]},
    config=config,
)
print("PAUSED. Current state:")
print(result["messages"][-1].content)   # shows the drafted email

# Step 2: Human reviews and (optionally) edits the draft
# Modify state if needed
app.update_state(config, {
    "messages": [AIMessage(content="DRAFT:\nHi team, the deployment is complete. Regards, Alice.")]
})

# Step 3: Resume — pass None as input to continue from breakpoint
final = app.invoke(None, config=config)
print(final["messages"][-1].content)   # "Email sent successfully."

interrupt_after — pause after a node completes

# Pause AFTER "draft" runs (so you can see what it produced before proceeding)
app = graph.compile(checkpointer=checkpointer, interrupt_after=["draft"])

Dynamic interrupts with interrupt()

from langgraph.types import interrupt, Command

def risky_action(state: MessagesState) -> dict:
    action_details = "Delete 500 rows from production database"

    # Pause and ask the human for approval
    decision = interrupt({
        "action": action_details,
        "prompt": "Do you approve this action? Reply 'yes' or 'no'.",
    })

    if decision.lower() == "yes":
        # proceed
        return {"messages": [AIMessage(content="Action completed.")]}
    else:
        return {"messages": [AIMessage(content="Action cancelled by user.")]}

# Resume by passing the human's response
final = app.invoke(
    Command(resume="yes"),   # human approved
    config=config,
)

Checking if execution was interrupted

snapshot = app.get_state(config)

# If the graph is paused, snapshot.next is non-empty
if snapshot.next:
    print(f"Graph paused. Next node(s) to run: {snapshot.next}")
    print("Current state:")
    for msg in snapshot.values["messages"]:
        print(f"  [{msg.type}] {msg.content[:80]}")
else:
    print("Graph completed.")

Real-world HITL example — content moderation

from typing import TypedDict

class ContentState(TypedDict):
    content: str
    flagged: bool
    approved: bool
    reason: str

def generate_content(state: ContentState) -> dict:
    resp = llm.invoke(f"Write a blog post about: {state['content']}")
    return {"content": resp.content, "flagged": False}

def auto_check(state: ContentState) -> dict:
    # Simple keyword check
    flagged = any(word in state["content"].lower() for word in ["spam", "scam"])
    return {"flagged": flagged}

def route_content(state: ContentState) -> str:
    return "human_review" if state["flagged"] else "publish"

def publish(state: ContentState) -> dict:
    return {"approved": True, "reason": "Auto-approved"}

g = StateGraph(ContentState)
g.add_node("generate",     generate_content)
g.add_node("auto_check",   auto_check)
g.add_node("publish",      publish)
g.add_edge(START, "generate")
g.add_edge("generate", "auto_check")
g.add_conditional_edges("auto_check", route_content, {
    "human_review": END,   # stop here — human reviews externally
    "publish": "publish",
})
g.add_edge("publish", END)

app = g.compile(checkpointer=MemorySaver())
Checkpointer required: HITL only works when a checkpointer is attached to the graph. Without it, the graph has no way to save state between the pause and the resume. Always use at least MemorySaver during development.
