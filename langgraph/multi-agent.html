Multi-Agent Systems

Multi-agent systems break complex tasks into specialised agents — each with its own tools and expertise — coordinated by a supervisor or through direct handoffs. LangGraph makes this natural: each agent is just a subgraph or a node, and the overall coordination is just another graph.

Patterns overview

Supervisor

  • One LLM routes tasks
  • Decides which agent runs next
  • Best for clear specialisations

Swarm / Handoff

  • Agents hand off to each other
  • Peer-to-peer routing
  • Best for sequential workflows

Subgraphs

  • Full graph as a single node
  • Independent state per subgraph
  • Best for modular reuse

Supervisor pattern

from typing import TypedDict, Annotated, List, Literal
from langgraph.graph import StateGraph, START, END, MessagesState
from langgraph.graph.message import add_messages
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, SystemMessage, BaseMessage
from langchain_core.tools import tool

llm = ChatOpenAI(model="gpt-4o", temperature=0)

# ── Worker agents ──
@tool
def search_web(query: str) -> str:
    """Search the web for information."""
    return f"Web result for '{query}': [simulated result]"

@tool
def run_python(code: str) -> str:
    """Execute Python code and return output."""
    try:
        exec_globals = {}
        exec(code, exec_globals)
        return str(exec_globals.get("result", "Done"))
    except Exception as e:
        return f"Error: {e}"

researcher_llm = llm.bind_tools([search_web])
coder_llm      = llm.bind_tools([run_python])

class SupervisorState(MessagesState):
    next_agent: str   # which agent to call next

# ── Supervisor node ──
AGENTS = ["researcher", "coder", "FINISH"]

def supervisor_node(state: SupervisorState) -> dict:
    system = SystemMessage(content=f"""You are a supervisor managing these agents: {AGENTS}.
Given the conversation, decide who acts next.
Reply with ONLY one of: researcher, coder, FINISH""")
    response = llm.invoke([system] + state["messages"])
    return {"next_agent": response.content.strip()}

# ── Worker nodes ──
def researcher_node(state: SupervisorState) -> dict:
    resp = researcher_llm.invoke(state["messages"])
    return {"messages": [resp]}

def coder_node(state: SupervisorState) -> dict:
    resp = coder_llm.invoke(state["messages"])
    return {"messages": [resp]}

# ── Graph ──
g = StateGraph(SupervisorState)
g.add_node("supervisor", supervisor_node)
g.add_node("researcher", researcher_node)
g.add_node("coder",      coder_node)

g.add_edge(START, "supervisor")
g.add_conditional_edges(
    "supervisor",
    lambda state: state["next_agent"],
    {"researcher": "researcher", "coder": "coder", "FINISH": END},
)
g.add_edge("researcher", "supervisor")   # report back to supervisor
g.add_edge("coder",      "supervisor")

app = g.compile()
result = app.invoke({"messages": [HumanMessage("Research Docker and write Python code to parse the docs.")]})
print(result["messages"][-1].content)

Agent handoff with Command

from langgraph.types import Command

# Command lets a node directly route to the next node
def triage_agent(state: MessagesState) -> Command:
    msg = state["messages"][-1].content.lower()

    if "search" in msg or "find" in msg:
        return Command(goto="researcher", update={"messages": state["messages"]})
    elif "code" in msg or "write" in msg:
        return Command(goto="coder", update={"messages": state["messages"]})
    else:
        return Command(goto=END, update={})

Subgraphs — agents as reusable building blocks

from langgraph.graph import StateGraph, START, END

# Build a self-contained research subgraph
class ResearchState(TypedDict):
    query: str
    findings: List[str]

def fetch_sources(state: ResearchState) -> dict:
    return {"findings": [f"Source 1 for {state['query']}", "Source 2"]}

def summarise(state: ResearchState) -> dict:
    summary = f"Summary: {'; '.join(state['findings'])}"
    return {"findings": [summary]}

research_graph = StateGraph(ResearchState)
research_graph.add_node("fetch", fetch_sources)
research_graph.add_node("summarise", summarise)
research_graph.add_edge(START, "fetch")
research_graph.add_edge("fetch", "summarise")
research_graph.add_edge("summarise", END)

research_app = research_graph.compile()

# Use the subgraph as a node inside the parent graph
class ParentState(TypedDict):
    task: str
    research_result: str

def run_research(state: ParentState) -> dict:
    result = research_app.invoke({"query": state["task"], "findings": []})
    return {"research_result": result["findings"][-1]}

parent = StateGraph(ParentState)
parent.add_node("research", run_research)
parent.add_edge(START, "research")
parent.add_edge("research", END)
parent_app = parent.compile()
When to use which pattern: Use the supervisor when task routing is complex and dynamic. Use handoffs / Command for linear pipelines where each agent knows who comes next. Use subgraphs when you want to reuse an agent in multiple parent graphs.
