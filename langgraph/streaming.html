Streaming & Async

LangGraph has built-in streaming support at multiple levels — node outputs, LLM token streams, and custom events. This enables responsive UIs that show progress as the graph executes rather than waiting for the entire run to finish.

Stream modes comparison

ModeYieldsBest for
"values"Full state after each nodeInspect complete state at each step
"updates"State delta from each nodeEfficient: only changed fields
"messages"LLM tokens as they generateLive chat UI, token-by-token display
"debug"All internal eventsDebugging, tracing
"custom"Your own events via astream_eventsCustom UI progress indicators

stream("values") — full state at each step

from langgraph.graph import StateGraph, START, END, MessagesState
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage

llm = ChatOpenAI(model="gpt-4o-mini")

def step1(state): return {"messages": [llm.invoke("Step 1: " + state["messages"][-1].content)]}
def step2(state): return {"messages": [llm.invoke("Step 2: Refine this: " + state["messages"][-1].content)]}

g = StateGraph(MessagesState)
g.add_node("s1", step1); g.add_node("s2", step2)
g.add_edge(START, "s1"); g.add_edge("s1", "s2"); g.add_edge("s2", END)
app = g.compile()

for state in app.stream(
    {"messages": [HumanMessage("Explain Docker")]},
    stream_mode="values",
):
    last = state["messages"][-1]
    print(f"[{last.type}] {last.content[:60]}...")

stream("updates") — delta per node

for update in app.stream(
    {"messages": [HumanMessage("Explain Kubernetes")]},
    stream_mode="updates",
):
    # update = {"node_name": {state_delta}}
    for node_name, state_delta in update.items():
        if "messages" in state_delta:
            msg = state_delta["messages"][-1]
            print(f"[{node_name}] → {msg.content[:80]}")

stream("messages") — token-by-token LLM output

from langchain_core.messages import AIMessageChunk

# Stream individual tokens as the LLM generates them
for chunk, metadata in app.stream(
    {"messages": [HumanMessage("Write a short poem about agents")]},
    stream_mode="messages",
):
    if isinstance(chunk, AIMessageChunk) and chunk.content:
        print(chunk.content, end="", flush=True)   # live token display

Async streaming — for FastAPI / web apps

import asyncio
from langchain_core.messages import HumanMessage

async def stream_response():
    async for chunk, metadata in app.astream(
        {"messages": [HumanMessage("List 5 Docker best practices")]},
        stream_mode="messages",
    ):
        if hasattr(chunk, "content") and chunk.content:
            print(chunk.content, end="", flush=True)

asyncio.run(stream_response())

# FastAPI SSE endpoint
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
import json

api = FastAPI()

@api.post("/chat")
async def chat_endpoint(user_input: str):
    async def generate():
        async for chunk, _ in app.astream(
            {"messages": [HumanMessage(user_input)]},
            stream_mode="messages",
        ):
            if hasattr(chunk, "content") and chunk.content:
                yield f"data: {json.dumps({'token': chunk.content})}\n\n"
        yield "data: [DONE]\n\n"

    return StreamingResponse(generate(), media_type="text/event-stream")

Multiple stream modes simultaneously

# Get both updates AND messages in one stream
for event in app.stream(
    {"messages": [HumanMessage("What is RAG?")]},
    stream_mode=["updates", "messages"],
):
    event_type, payload = event
    if event_type == "updates":
        print(f"NODE UPDATE: {list(payload.keys())}")
    elif event_type == "messages":
        chunk, metadata = payload
        if hasattr(chunk, "content") and chunk.content:
            print(chunk.content, end="", flush=True)

astream_events — rich event stream

async def watch_events():
    async for event in app.astream_events(
        {"messages": [HumanMessage("Summarise LangGraph")]},
        version="v2",
    ):
        kind = event["event"]

        if kind == "on_chat_model_stream":
            chunk = event["data"]["chunk"]
            print(chunk.content, end="", flush=True)

        elif kind == "on_chain_start":
            print(f"\n[START] {event['name']}")

        elif kind == "on_chain_end":
            print(f"\n[END]   {event['name']}")

        elif kind == "on_tool_start":
            print(f"\n[TOOL]  {event['name']}({event['data'].get('input', {})})")

asyncio.run(watch_events())
Production tip: In FastAPI or Django Channels, use astream or astream_events with Server-Sent Events (SSE) to push tokens to the browser in real time. Set streaming=True on your ChatOpenAI to ensure tokens flow through; without it, the LLM buffers everything and the stream sends one large chunk at the end.
