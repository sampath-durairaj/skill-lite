Nodes & Edges

Nodes are Python functions that process state. Edges are the connections between them — either fixed (always go to node B after A) or conditional (decide at runtime which node to go to next based on state). Together they define your workflow's logic.

Nodes — the workhorses

from typing import TypedDict, Annotated, List
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_core.messages import BaseMessage, AIMessage, HumanMessage
from langchain_openai import ChatOpenAI

class State(TypedDict):
    messages: Annotated[List[BaseMessage], add_messages]
    tool_result: str

llm = ChatOpenAI(model="gpt-4o-mini")

# Simple node — transform state
def llm_node(state: State) -> dict:
    response = llm.invoke(state["messages"])
    return {"messages": [response]}

# Node with conditional logic
def validate_node(state: State) -> dict:
    last_msg = state["messages"][-1].content if state["messages"] else ""
    if len(last_msg) < 10:
        return {"tool_result": "too_short"}
    return {"tool_result": "ok"}

# Node that calls tools
from langchain_core.tools import tool

@tool
def search(query: str) -> str:
    """Search for information."""
    return f"Results for: {query}"

tools = [search]
llm_with_tools = llm.bind_tools(tools)

def agent_node(state: State) -> dict:
    response = llm_with_tools.invoke(state["messages"])
    return {"messages": [response]}

Adding nodes to the graph

graph = StateGraph(State)

# Add node with a name
graph.add_node("llm", llm_node)
graph.add_node("validate", validate_node)
graph.add_node("agent", agent_node)

# You can also add a lambda inline
graph.add_node("uppercase", lambda state: {
    "tool_result": state["tool_result"].upper()
})

Fixed edges — deterministic flow

# Always go from A → B
graph.add_edge(START, "llm")       # entry point
graph.add_edge("llm", "validate")  # then validate
graph.add_edge("validate", END)    # then exit

# Multiple out-edges: both B and C run after A (parallel)
graph.add_edge("preprocess", "branch_a")
graph.add_edge("preprocess", "branch_b")

Conditional edges — dynamic routing

# Router function: reads state, returns the NEXT NODE NAME (as a string)
def should_continue(state: State) -> str:
    last_message = state["messages"][-1]

    # If the LLM returned tool calls, run the tool executor
    if hasattr(last_message, "tool_calls") and last_message.tool_calls:
        return "tools"

    # Otherwise we're done
    return END

# Add a conditional edge FROM "agent"
# The router function decides WHICH node to go to
graph.add_conditional_edges("agent", should_continue)

# You can also map return values to node names explicitly
graph.add_conditional_edges(
    "validate",
    lambda state: state["tool_result"],
    {
        "ok":        "proceed",   # if result == "ok" → proceed node
        "too_short": "retry",     # if result == "too_short" → retry node
        "error":     END,         # if result == "error" → exit
    }
)

Complete ReAct agent loop

import json
from langchain_core.messages import ToolMessage

class AgentState(TypedDict):
    messages: Annotated[List[BaseMessage], add_messages]

# Node 1: LLM decides what to do
def call_llm(state: AgentState) -> dict:
    response = llm_with_tools.invoke(state["messages"])
    return {"messages": [response]}

# Node 2: Execute tool calls from the LLM
def call_tools(state: AgentState) -> dict:
    last_msg = state["messages"][-1]
    tool_map = {t.name: t for t in tools}
    results = []
    for tc in last_msg.tool_calls:
        result = tool_map[tc["name"]].invoke(tc["args"])
        results.append(ToolMessage(content=str(result), tool_call_id=tc["id"]))
    return {"messages": results}

# Router: tool calls present → call_tools; else → END
def route(state: AgentState) -> str:
    last = state["messages"][-1]
    if hasattr(last, "tool_calls") and last.tool_calls:
        return "call_tools"
    return END

# Build graph
g = StateGraph(AgentState)
g.add_node("llm",  call_llm)
g.add_node("tools", call_tools)
g.add_edge(START, "llm")
g.add_conditional_edges("llm", route)
g.add_edge("tools", "llm")   # ← CYCLE: after tools, ask LLM again

app = g.compile()
result = app.invoke({"messages": [HumanMessage(content="Search for LangGraph docs")]})
print(result["messages"][-1].content)

Entry and finish points

# Default entry is START; override with set_entry_point
graph.set_entry_point("my_first_node")  # same as add_edge(START, "my_first_node")

# Default finish is END; also set conditionally
graph.set_finish_point("my_last_node")  # same as add_edge("my_last_node", END)

# Multiple entry/exit paths are fine
graph.add_edge(START, "node_a")
graph.add_edge(START, "node_b")   # both start simultaneously
Cycle design: When building a loop (agent → tools → agent), always have an exit condition in your conditional edge. Use a loop counter in state and route to END after max_iterations to prevent runaway graphs and runaway API costs.
