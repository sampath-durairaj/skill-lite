Checkpointing

Checkpointing saves the graph's state after every step into a store. This enables three key capabilities: multi-turn conversations (pick up where you left off), time-travel debugging (replay or branch from any past checkpoint), and human-in-the-loop (pause execution and resume after approval).

MemorySaver — in-memory (development)

from langgraph.graph import StateGraph, START, END, MessagesState
from langgraph.checkpoint.memory import MemorySaver
from langchain_openai import ChatOpenAI

llm = ChatOpenAI(model="gpt-4o-mini")

def chat(state: MessagesState) -> dict:
    return {"messages": [llm.invoke(state["messages"])]}

graph = StateGraph(MessagesState)
graph.add_node("chat", chat)
graph.add_edge(START, "chat")
graph.add_edge("chat", END)

# Attach a checkpointer
checkpointer = MemorySaver()
app = graph.compile(checkpointer=checkpointer)

# config MUST include thread_id to identify the session
config = {"configurable": {"thread_id": "user_alice"}}

from langchain_core.messages import HumanMessage

# Turn 1
r1 = app.invoke({"messages": [HumanMessage("My name is Alice.")]}, config=config)
print(r1["messages"][-1].content)

# Turn 2 — state is automatically restored; no need to re-send history
r2 = app.invoke({"messages": [HumanMessage("What is my name?")]}, config=config)
print(r2["messages"][-1].content)   # "Your name is Alice."

SqliteSaver — persistent across restarts

pip install langgraph-checkpoint-sqlite
from langgraph.checkpoint.sqlite import SqliteSaver

# State survives process restarts
with SqliteSaver.from_conn_string("checkpoints.db") as checkpointer:
    app = graph.compile(checkpointer=checkpointer)

    config = {"configurable": {"thread_id": "session_42"}}
    app.invoke({"messages": [HumanMessage("Start a new task.")]}, config=config)

    # Later — even after restart — resume the same session
    app.invoke({"messages": [HumanMessage("Continue where we left off.")]}, config=config)

Reading state snapshots

# Get the current state of a thread
snapshot = app.get_state(config)

print(snapshot.values)      # full state dict
print(snapshot.next)        # nodes that will run next (empty if done)
print(snapshot.config)      # config with checkpoint_id
print(snapshot.metadata)    # step number, run_id, parent_id

# List all checkpoints for a thread (most recent first)
history = list(app.get_state_history(config))
for checkpoint in history:
    print(f"Step {checkpoint.metadata['step']}: {checkpoint.next}")

Time travel — replay from a past checkpoint

# Get the checkpoint ID from history
history = list(app.get_state_history(config))
step_2_checkpoint = history[-3]   # 3rd-from-last = step 2

# Replay from that checkpoint — creates a new branch
past_config = step_2_checkpoint.config
result = app.invoke(
    {"messages": [HumanMessage("Try a different approach.")]},
    config=past_config,
)
# This forks the thread — the original thread is unchanged

Manually updating state

from langchain_core.messages import AIMessage

# Override part of the state without running any nodes
app.update_state(
    config,
    {"messages": [AIMessage(content="Let me correct that answer: 42.")]},
    as_node="chat",   # pretend this update came from the "chat" node
)

# Now continue from the corrected state
app.invoke({"messages": [HumanMessage("Thanks! What's next?")]}, config=config)

PostgresSaver — production-grade persistence

pip install langgraph-checkpoint-postgres psycopg
from langgraph.checkpoint.postgres import PostgresSaver

DB_URI = "postgresql://user:password@localhost:5432/mydb"

with PostgresSaver.from_conn_string(DB_URI) as checkpointer:
    checkpointer.setup()   # creates tables on first run
    app = graph.compile(checkpointer=checkpointer)
    # ... production usage

Checkpointer comparison

CheckpointerPersistenceBest for
MemorySaverProcess memory onlyDevelopment & testing
SqliteSaverSQLite file on diskSingle-server apps, prototypes
PostgresSaverPostgreSQL databaseProduction, multi-server
RedisSaverRedis (in-memory + TTL)High-throughput, short-lived sessions
thread_id is your session key. Every distinct conversation or user session needs a unique thread_id. Use str(uuid.uuid4()) for new sessions and store the ID client-side (cookie / localStorage) to resume them. The same graph compiled with different checkpointers is a common pattern: MemorySaver in tests, PostgresSaver in production.
