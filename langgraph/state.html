State Management

State is the shared data structure that flows through every node in a LangGraph graph. Nodes read from it and return partial updates; LangGraph merges those updates into the running state automatically. Getting state design right is the most important part of building a reliable graph.

Defining state with TypedDict

from typing import TypedDict, List, Optional

class AgentState(TypedDict):
    # Input
    user_input: str

    # Working memory
    messages: List[dict]         # conversation history
    tool_calls: List[str]        # tools invoked so far
    iteration: int               # loop counter

    # Output
    final_answer: Optional[str]
    error: Optional[str]

Annotated reducers — controlling how state is merged

By default, each key in state is overwritten by a node's return value. Use Annotated with a reducer function to change how updates are merged — for example, appending to a list instead of replacing it.

from typing import Annotated
from langgraph.graph.message import add_messages
from langchain_core.messages import BaseMessage
import operator

class ChatState(TypedDict):
    # add_messages reducer: APPENDS new messages, never replaces
    messages: Annotated[List[BaseMessage], add_messages]

    # operator.add reducer: APPENDS list items
    tool_results: Annotated[List[str], operator.add]

    # Default (no Annotated): OVERWRITES on each update
    current_step: str
    is_done: bool

The add_messages reducer (most common)

from typing import Annotated, List
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_core.messages import HumanMessage, AIMessage, BaseMessage
from langchain_openai import ChatOpenAI

class State(TypedDict):
    messages: Annotated[List[BaseMessage], add_messages]

llm = ChatOpenAI(model="gpt-4o-mini")

def chat_node(state: State) -> dict:
    response = llm.invoke(state["messages"])
    return {"messages": [response]}   # add_messages APPENDS this

graph = StateGraph(State)
graph.add_node("chat", chat_node)
graph.add_edge(START, "chat")
graph.add_edge("chat", END)
app = graph.compile()

# First turn
result = app.invoke({"messages": [HumanMessage(content="What is LangGraph?")]})
# messages now = [HumanMessage(...), AIMessage(...)]

# Second turn — append to existing messages
result2 = app.invoke({
    "messages": result["messages"] + [HumanMessage(content="Give me an example.")]
})

MessagesState — built-in shortcut

from langgraph.graph import MessagesState   # ready-made state with add_messages

# Equivalent to:
# class MessagesState(TypedDict):
#     messages: Annotated[List[BaseMessage], add_messages]

class MyState(MessagesState):
    # Extend with extra fields
    user_id: str
    session_count: int

Custom reducer functions

def keep_last_5(existing: List[str], new: List[str]) -> List[str]:
    """Keep only the 5 most recent items."""
    combined = existing + new
    return combined[-5:]

class State(TypedDict):
    # Custom reducer: sliding window of last 5 items
    recent_thoughts: Annotated[List[str], keep_last_5]

    # Reducer that sums integers
    total_tokens: Annotated[int, operator.add]

    # Reducer that keeps the maximum value
    best_score: Annotated[float, max]

Reading and updating state in nodes

def process_node(state: State) -> dict:
    # Read any field directly
    user_msg = state["user_input"]
    count = state["iteration"]
    history = state["messages"]

    # Only return the fields you want to change
    # (omitted fields stay unchanged)
    return {
        "iteration": count + 1,
        "messages": [AIMessage(content="Processing...")],
        # "user_input" is NOT returned — stays the same
    }

# Nodes can also return None to make no state change
def check_node(state: State) -> dict | None:
    if state["is_done"]:
        return None   # no-op
    return {"current_step": "running"}

Accessing the full state snapshot

from langgraph.checkpoint.memory import MemorySaver

checkpointer = MemorySaver()
app = graph.compile(checkpointer=checkpointer)

config = {"configurable": {"thread_id": "session_1"}}

# Run the graph
app.invoke({"user_input": "Hello", "messages": [], "iteration": 0}, config=config)

# Get a snapshot of the current state
snapshot = app.get_state(config)
print(snapshot.values)         # full state dict
print(snapshot.next)           # which node runs next (empty if done)
print(snapshot.metadata)       # step count, run id, etc.

# Update state manually (useful for testing or HITL)
app.update_state(config, {"iteration": 0, "error": None})
State design rule: Start minimal — only add fields a node actually needs to pass to another node. Don't use state as a general scratchpad. Use the add_messages reducer for any message list so concurrent nodes can both add messages without one overwriting the other.
