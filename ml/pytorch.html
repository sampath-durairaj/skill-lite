PyTorch

PyTorch is the dominant deep learning framework — used by most research labs and increasingly in production. Its define-by-run approach (dynamic computation graph) makes debugging natural and enables complex architectures that are hard to express in static-graph frameworks.

Tensors — the building block

import torch

# Create tensors
a = torch.tensor([[1.0, 2.0], [3.0, 4.0]])
b = torch.zeros(2, 3)
c = torch.ones(4)
d = torch.randn(3, 3)           # standard normal
e = torch.arange(0, 10, 2)     # [0, 2, 4, 6, 8]

# Device
device = "cuda" if torch.cuda.is_available() else "cpu"
a = a.to(device)

# Shape manipulation
print(a.shape)                  # torch.Size([2, 2])
print(a.reshape(1, 4))
print(a.T)                      # transpose
print(a.flatten())

# NumPy bridge (CPU tensors only)
import numpy as np
n = a.cpu().numpy()
t = torch.from_numpy(n)         # shares memory

Autograd — automatic differentiation

import torch

x = torch.tensor(3.0, requires_grad=True)
y = x ** 2 + 2 * x + 1   # y = (x+1)²

y.backward()              # compute dy/dx
print(x.grad)             # dy/dx = 2x + 2 = 8.0

# Multi-variable
x = torch.tensor([1.0, 2.0, 3.0], requires_grad=True)
loss = (x ** 2).sum()
loss.backward()
print(x.grad)             # [2, 4, 6] — gradient of each element

# No grad during inference (saves memory + compute)
with torch.no_grad():
    pred = model(x_input)

Building a model with nn.Module

import torch
import torch.nn as nn
import torch.nn.functional as F

class MLP(nn.Module):
    def __init__(self, input_dim, hidden_dim, output_dim, dropout=0.3):
        super().__init__()
        self.fc1 = nn.Linear(input_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, hidden_dim // 2)
        self.fc3 = nn.Linear(hidden_dim // 2, output_dim)
        self.dropout = nn.Dropout(dropout)
        self.bn1 = nn.BatchNorm1d(hidden_dim)
        self.bn2 = nn.BatchNorm1d(hidden_dim // 2)

    def forward(self, x):
        x = F.relu(self.bn1(self.fc1(x)))
        x = self.dropout(x)
        x = F.relu(self.bn2(self.fc2(x)))
        x = self.dropout(x)
        return self.fc3(x)   # logits (no softmax — CrossEntropyLoss handles it)

model = MLP(input_dim=20, hidden_dim=128, output_dim=5)
print(model)
print(f"Parameters: {sum(p.numel() for p in model.parameters()):,}")

Complete training loop

import torch
import torch.nn as nn
from torch.utils.data import DataLoader, TensorDataset
from sklearn.datasets import make_classification
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
import numpy as np

# Data
X, y = make_classification(n_samples=2000, n_features=20, n_classes=3,
                            n_informative=10, random_state=42)
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test  = scaler.transform(X_test)

# Tensors
X_tr = torch.tensor(X_train, dtype=torch.float32)
y_tr = torch.tensor(y_train, dtype=torch.long)
X_te = torch.tensor(X_test,  dtype=torch.float32)
y_te = torch.tensor(y_test,  dtype=torch.long)

# DataLoader
train_ds = TensorDataset(X_tr, y_tr)
train_dl = DataLoader(train_ds, batch_size=64, shuffle=True)

device = "cuda" if torch.cuda.is_available() else "cpu"
model  = MLP(20, 128, 3).to(device)
criterion = nn.CrossEntropyLoss()
optimizer = torch.optim.Adam(model.parameters(), lr=1e-3, weight_decay=1e-4)
scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=50)

# Training loop
for epoch in range(50):
    model.train()
    total_loss = 0
    for xb, yb in train_dl:
        xb, yb = xb.to(device), yb.to(device)
        optimizer.zero_grad()
        logits = model(xb)
        loss   = criterion(logits, yb)
        loss.backward()
        optimizer.step()
        total_loss += loss.item()
    scheduler.step()

    if (epoch + 1) % 10 == 0:
        model.eval()
        with torch.no_grad():
            test_logits = model(X_te.to(device))
            test_acc    = (test_logits.argmax(1) == y_te.to(device)).float().mean()
        print(f"Epoch {epoch+1:3d} | loss={total_loss/len(train_dl):.4f} | test_acc={test_acc:.3f}")

Save & load models

# Save — state dict only (recommended)
torch.save(model.state_dict(), "model.pt")

# Load
loaded = MLP(20, 128, 3)
loaded.load_state_dict(torch.load("model.pt", map_location="cpu"))
loaded.eval()

# Save full model (fragile — tied to code structure)
torch.save(model, "full_model.pt")

# Export to ONNX for deployment
dummy = torch.randn(1, 20)
torch.onnx.export(model.cpu(), dummy, "model.onnx",
                  input_names=["input"], output_names=["logits"])

Transfer learning

import torchvision.models as models

# Load pretrained ResNet-18
backbone = models.resnet18(weights="IMAGENET1K_V1")

# Freeze all layers
for param in backbone.parameters():
    param.requires_grad = False

# Replace final layer for your number of classes
n_classes = 5
backbone.fc = nn.Linear(backbone.fc.in_features, n_classes)

# Only the new head trains
optimizer = torch.optim.Adam(backbone.fc.parameters(), lr=1e-3)
Training tips: (1) Use model.train() before training and model.eval() before inference — this toggles Dropout and BatchNorm behaviour. (2) Always call optimizer.zero_grad() before loss.backward(). (3) Use a learning rate scheduler — CosineAnnealingLR or ReduceLROnPlateau consistently improve final accuracy.
