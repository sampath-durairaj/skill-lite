Feature Engineering

Feature engineering is the process of transforming raw data into the numerical representations that ML models can learn from. It often has a larger impact on model performance than algorithm choice — garbage in, garbage out, no matter how sophisticated the model.

Scaling numerical features

from sklearn.preprocessing import (
    StandardScaler, MinMaxScaler, RobustScaler, MaxAbsScaler
)
import numpy as np

X = np.array([[100, 0.01], [200, 0.02], [150, 0.015], [300, 0.005]])

# StandardScaler: mean=0, std=1 — use when normal distribution assumed
std   = StandardScaler().fit_transform(X)

# MinMaxScaler: [0, 1] — use when you know the bounds
mms   = MinMaxScaler().fit_transform(X)

# RobustScaler: median/IQR — use when outliers are present
rob   = RobustScaler().fit_transform(X)

# When to scale?
# Required:  LogReg, SVM, KNN, Neural Networks, Ridge/Lasso, PCA
# Optional:  Random Forest, Decision Trees, XGBoost (tree-based — scale invariant)

Encoding categorical features

import pandas as pd
from sklearn.preprocessing import OrdinalEncoder, OneHotEncoder
from sklearn.pipeline import Pipeline
from sklearn.compose import ColumnTransformer

df = pd.DataFrame({
    "size":   ["S", "M", "L", "XL", "M"],      # ordinal
    "colour": ["red", "blue", "green", "red", "blue"],  # nominal
    "price":  [10.0, 15.0, 20.0, 25.0, 15.0],
})

# Ordinal encoding — preserves order
ord_enc = OrdinalEncoder(categories=[["S", "M", "L", "XL"]])
df["size_enc"] = ord_enc.fit_transform(df[["size"]])

# One-hot encoding — use for nominal (no order) with low cardinality
ohe = OneHotEncoder(sparse_output=False, drop="first")   # drop first avoids multicollinearity
ohe_result = ohe.fit_transform(df[["colour"]])

# ColumnTransformer — apply different transforms to different columns
from sklearn.preprocessing import StandardScaler
ct = ColumnTransformer([
    ("num",  StandardScaler(),           ["price"]),
    ("ord",  OrdinalEncoder(categories=[["S","M","L","XL"]]), ["size"]),
    ("cat",  OneHotEncoder(drop="first"), ["colour"]),
])
X_transformed = ct.fit_transform(df.drop(columns=[]))  # returns ndarray

Handling missing values

import pandas as pd
import numpy as np
from sklearn.impute import SimpleImputer, KNNImputer

df = pd.DataFrame({
    "age":    [25, np.nan, 35, np.nan, 45],
    "income": [50000, 60000, np.nan, 80000, np.nan],
    "gender": ["M", "F", np.nan, "M", "F"],
})

# Strategy choice
# Numerical: mean (normal), median (skewed/outliers), constant
# Categorical: most_frequent, constant

num_imp = SimpleImputer(strategy="median")
cat_imp = SimpleImputer(strategy="most_frequent")

df["age"]    = num_imp.fit_transform(df[["age"]])
df["income"] = num_imp.fit_transform(df[["income"]])
df["gender"] = cat_imp.fit_transform(df[["gender"]])

# KNN imputer — uses nearest neighbours (more accurate, slower)
knn_imp = KNNImputer(n_neighbors=3)
df_imputed = pd.DataFrame(
    knn_imp.fit_transform(df[["age", "income"]]),
    columns=["age", "income"]
)

# Always add a "was_missing" indicator feature
df["income_missing"] = df["income"].isna().astype(int)

scikit-learn Pipeline — production-safe preprocessing

from sklearn.pipeline import Pipeline
from sklearn.compose import ColumnTransformer
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.ensemble import GradientBoostingClassifier
from sklearn.model_selection import cross_val_score

# Define feature groups
numeric_features     = ["age", "income", "tenure"]
categorical_features = ["gender", "product", "region"]

# Preprocessing for numeric features
numeric_transformer = Pipeline([
    ("imputer", SimpleImputer(strategy="median")),
    ("scaler",  StandardScaler()),
])

# Preprocessing for categorical features
categorical_transformer = Pipeline([
    ("imputer", SimpleImputer(strategy="most_frequent")),
    ("onehot",  OneHotEncoder(handle_unknown="ignore")),
])

# Combine
preprocessor = ColumnTransformer([
    ("num", numeric_transformer,     numeric_features),
    ("cat", categorical_transformer, categorical_features),
])

# Full pipeline — preprocessing + model
full_pipeline = Pipeline([
    ("preprocessor", preprocessor),
    ("model",        GradientBoostingClassifier(n_estimators=100)),
])

# Now everything is safe to cross-validate (no data leakage)

rf.fit(X_train, y_train)
sfm = SelectFromModel(rf, threshold="median")
X_sfm = sfm.fit_transform(X_train, y_train)
print(f"Features kept: {X_sfm.shape[1]} / {X_train.shape[1]}")

# Wrapper: recursive feature elimination with CV (slow but thorough)
rfecv = RFECV(
    estimator=RandomForestClassifier(n_estimators=50, random_state=42),
    step=1, cv=5, scoring="roc_auc", n_jobs=-1
)
rfecv.fit(X_train, y_train)
print(f"Optimal features: {rfecv.n_features_}")
Pipeline is mandatory: Always wrap preprocessing steps in a Pipeline. This prevents data leakage — if you scale the full dataset before splitting, your test set statistics contaminate the training process, causing falsely optimistic evaluation scores. The Pipeline ensures fit only happens on training data during cross-validation.
