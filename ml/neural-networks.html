Neural Networks

Neural networks are universal function approximators built from layers of interconnected neurons. Each neuron computes a weighted sum of its inputs and passes it through a non-linear activation function. Stacking many layers creates a deep network that can learn extraordinarily complex representations.

Anatomy of a neuron

import numpy as np

def neuron(inputs, weights, bias, activation="relu"):
    z = np.dot(weights, inputs) + bias   # linear combination
    if activation == "relu":    return max(0, z)
    if activation == "sigmoid": return 1 / (1 + np.exp(-z))
    if activation == "tanh":    return np.tanh(z)
    return z   # linear (no activation)

# Example: single neuron with 3 inputs
inputs  = np.array([0.5, -1.2, 0.8])
weights = np.array([0.4,  0.3, -0.6])
bias    = 0.1
print(neuron(inputs, weights, bias))   # relu(0.5*0.4 + (-1.2)*0.3 + 0.8*(-0.6) + 0.1)

MLP with scikit-learn

from sklearn.neural_network import MLPClassifier
from sklearn.datasets import load_digits
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler

digits = load_digits()
X_train, X_test, y_train, y_test = train_test_split(
    digits.data, digits.target, test_size=0.2, random_state=42
)

scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test  = scaler.transform(X_test)

mlp = MLPClassifier(
    hidden_layer_sizes=(256, 128),   # two hidden layers
    activation="relu",
    solver="adam",
    learning_rate_init=1e-3,
    max_iter=300,
    early_stopping=True,             # stop when val loss stops improving
    validation_fraction=0.1,
    random_state=42,
)
mlp.fit(X_train, y_train)
print(f"Test accuracy: {mlp.score(X_test, y_test):.3f}")
print(f"Loss curve: {len(mlp.loss_curve_)} epochs")

Backpropagation — the algorithm

import numpy as np

# Tiny 2-layer network: input(2) → hidden(3) → output(1)
def sigmoid(z): return 1 / (1 + np.exp(-z))
def sigmoid_prime(z): s = sigmoid(z); return s * (1 - s)

# Random weights
rng = np.random.default_rng(0)
W1 = rng.standard_normal((3, 2)) * 0.1   # hidden: 3 neurons, 2 inputs
b1 = np.zeros((3, 1))
W2 = rng.standard_normal((1, 3)) * 0.1   # output: 1 neuron, 3 inputs
b2 = np.zeros((1, 1))

lr = 0.1
X = np.array([[0,0],[0,1],[1,0],[1,1]]).T   # XOR input  (2, 4)
y = np.array([[0,1,1,0]])                   # XOR output (1, 4)

for epoch in range(5000):
    # ── Forward pass ──
    z1 = W1 @ X + b1      # (3, 4)
    a1 = sigmoid(z1)
    z2 = W2 @ a1 + b2     # (1, 4)
    a2 = sigmoid(z2)

    # ── Loss (MSE) ──
    loss = ((a2 - y) ** 2).mean()

    # ── Backward pass ──
    dz2 = 2 * (a2 - y) * sigmoid_prime(z2) / y.shape[1]
    dW2 = dz2 @ a1.T
    db2 = dz2.sum(axis=1, keepdims=True)

    dz1 = (W2.T @ dz2) * sigmoid_prime(z1)
    dW1 = dz1 @ X.T
    db1 = dz1.sum(axis=1, keepdims=True)

    # ── Gradient descent ──
    W1 -= lr * dW1; b1 -= lr * db1
    W2 -= lr * dW2; b2 -= lr * db2

    if epoch % 1000 == 0:
        print(f"Epoch {epoch:5d}  loss={loss:.5f}")

print("XOR predictions:", a2.round(2))

Regularisation techniques

import torch
import torch.nn as nn

class RegularisedNet(nn.Module):
    def __init__(self, input_dim, hidden_dim, output_dim, dropout=0.3):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.BatchNorm1d(hidden_dim),   # normalise activations
            nn.ReLU(),
            nn.Dropout(dropout),          # randomly zero p fraction of neurons
            nn.Linear(hidden_dim, hidden_dim // 2),
            nn.BatchNorm1d(hidden_dim // 2),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(hidden_dim // 2, output_dim),
        )

    def forward(self, x):
        return self.net(x)

# L2 regularisation = weight_decay in the optimizer
optimizer = torch.optim.Adam(
    RegularisedNet(10, 128, 2).parameters(),
    lr=1e-3, weight_decay=1e-4
)
Architecture defaults: ReLU activations + Adam optimizer + BatchNorm + Dropout(0.2–0.5) is a solid default. Add more layers/neurons and increase dropout if you overfit; reduce model size if training is too slow. Always start with a simple baseline before going deep.
