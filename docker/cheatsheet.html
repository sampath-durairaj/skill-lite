Docker Cheat Sheet

A complete reference of the most-used Docker CLI commands, organised by category.

Container lifecycle

# Run a container
docker run -d -p 8080:80 --name web nginx
docker run -it --rm alpine sh                  # interactive, auto-remove on exit

# Start / stop / restart
docker start web
docker stop web                                # SIGTERM → SIGKILL after 10s
docker kill web                                # SIGKILL immediately
docker restart web

# Pause / unpause
docker pause web
docker unpause web

# Remove
docker rm web                                  # stopped container
docker rm -f web                               # running container (force)
docker container prune                         # all stopped containers

Listing & inspecting containers

docker ps                                      # running
docker ps -a                                   # all (incl. stopped)
docker ps -q                                   # quiet — IDs only
docker inspect web                             # full JSON metadata
docker inspect web --format '{{.NetworkSettings.IPAddress}}'
docker top web                                 # processes inside
docker stats                                   # live CPU / memory
docker port web                                # published port mappings

Logs & exec

docker logs web                                # print logs
docker logs -f web                             # follow (tail -f style)
docker logs --tail 100 web                     # last 100 lines
docker logs --since 30m web                    # last 30 minutes

docker exec -it web bash                       # open shell
docker exec web cat /etc/nginx/nginx.conf      # single command

docker cp web:/etc/nginx/nginx.conf ./         # container → host
docker cp ./nginx.conf web:/etc/nginx/         # host → container

Images

docker images                                  # list local images
docker pull nginx:1.27-alpine                  # pull from registry
docker push myuser/myapp:1.0                   # push to registry

docker build -t myapp:1.0 .                    # build from Dockerfile
docker build -f deploy/Dockerfile -t myapp .   # custom Dockerfile
docker build --build-arg ENV=prod -t myapp .   # with build arg

docker tag myapp:1.0 myuser/myapp:1.0         # tag
docker rmi myapp:1.0                           # remove image
docker image prune                             # remove dangling images
docker image prune -a                          # remove all unused images

docker image history myapp:1.0                 # layer history
docker image inspect myapp:1.0                 # metadata

Volumes

docker volume create pgdata
docker volume ls
docker volume inspect pgdata
docker volume rm pgdata
docker volume prune                            # unused volumes

# Mount in docker run
docker run -v pgdata:/var/lib/postgresql/data postgres
docker run -v "$(pwd)":/app -w /app node:20 npm start
docker run --mount type=volume,src=pgdata,target=/data postgres

Networks

docker network create mynet
docker network ls
docker network inspect mynet
docker network rm mynet
docker network prune

docker network connect mynet web              # connect running container
docker network disconnect mynet web

# Run on a network
docker run --network mynet --name app myimage

Registry

docker login                                   # Docker Hub
docker login ghcr.io                           # GitHub Container Registry
docker logout

docker pull nginx                              # latest tag
docker pull nginx:1.27-alpine                  # specific tag
docker pull nginx@sha256:abc123...             # by digest (immutable)

docker push myuser/myapp:1.0
docker search nginx                            # search Docker Hub

Docker Compose

docker compose up -d                           # start all, detached
docker compose up -d --build                   # rebuild images first
docker compose down                            # stop & remove containers
docker compose down -v                         # also remove volumes
docker compose ps                              # list service containers
docker compose logs -f                         # all service logs
docker compose logs -f web                     # one service
docker compose exec web bash                   # shell in service
docker compose run --rm web npm run migrate    # one-off command
docker compose pull                            # pull latest images
docker compose config                          # validate & print resolved config

System & cleanup

docker info                                    # daemon info
docker version                                 # client & daemon version
docker events                                  # live event stream

docker system df                               # disk usage summary
docker system prune                            # stopped containers, dangling images, unused networks
docker system prune -a                         # + all unused images
docker system prune -a --volumes               # + volumes (destructive!)

Dockerfile quick reference

FROM node:20-alpine          # base image
WORKDIR /app                 # set working dir
COPY package*.json ./        # copy files
RUN npm ci                   # run a command (creates a layer)
COPY . .                     # copy source
ARG BUILD_ENV                # build-time variable
ENV NODE_ENV=production      # runtime env variable
EXPOSE 3000                  # document port (informational)
USER node                    # switch user
HEALTHCHECK --interval=30s CMD wget -qO- http://localhost:3000/health || exit 1
CMD ["node", "server.js"]    # default command
ENTRYPOINT ["docker-entrypoint.sh"]  # main executable

Useful one-liners

# Remove all stopped containers
docker rm $(docker ps -aq -f status=exited)

# Remove all untagged images
docker rmi $(docker images -q -f dangling=true)

# Get a container's IP address
docker inspect -f '{{range .NetworkSettings.Networks}}{{.IPAddress}}{{end}}' web

# Enter a container as root (even if USER is set)
docker exec -u root -it web bash

# Watch container resource usage
watch -n1 docker stats --no-stream

# Export a container filesystem to tar
docker export web -o web-snapshot.tar

# Save an image to tar (shareable without registry)
docker save myapp:1.0 | gzip > myapp.tar.gz

# Load a saved image
docker load < myapp.tar.gz
