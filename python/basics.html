Python Basics

The core language concepts you'll use in every Python program: variables and data types, control flow, functions, error handling, and object-oriented programming. This page is the foundation for the NumPy, Pandas, and Matplotlib pages that follow.

Variables and data types

Python is dynamically typed — you don't declare a type, and a variable can be reassigned to a different type at any time.

age = 30                 # int
price = 19.99            # float
name = "Ada"              # str
is_active = True          # bool
tags = ["python", "dev"]  # list — ordered, mutable
point = (3, 4)             # tuple — ordered, immutable
user = {"name": "Ada", "age": 30}  # dict — key/value pairs
unique_ids = {1, 2, 3}     # set — unordered, no duplicates
nothing = None             # absence of a value

print(type(age), type(name), type(tags))

Operators

Control flow

score = 82

if score >= 90:
    grade = "A"
elif score >= 80:
    grade = "B"
else:
    grade = "C"

for i in range(5):        # 0, 1, 2, 3, 4
    print(i)

for tag in ["python", "numpy", "pandas"]:
    print(tag.upper())

n = 5
while n > 0:
    n -= 1                 # no ++/-- operators in Python

for i in range(10):
    if i == 3:
        continue           # skip this iteration
    if i == 7:
        break               # exit the loop
    print(i)

Strings and f-strings

name, age = "Ada", 30

greeting = f"{name} is {age} years old"      # f-string — preferred
greeting2 = "{} is {} years old".format(name, age)
greeting3 = "%s is %d years old" % (name, age)

f"{3.14159:.2f}"      # '3.14'  — 2 decimal places
f"{age:>5}"            # '   30' — right-aligned, width 5

"hello world".upper()       # 'HELLO WORLD'
"  hello  ".strip()          # 'hello'
"a,b,c".split(",")            # ['a', 'b', 'c']
",".join(["a", "b", "c"])      # 'a,b,c'
"hello"[1:4]                    # 'ell' — slicing

Functions

def greet(name, greeting="Hello"):   # default argument
    return f"{greeting}, {name}!"

greet("Ada")                    # "Hello, Ada!"
greet("Ada", greeting="Hi")     # "Hi, Ada!"

def total(*args, **kwargs):     # *args → tuple, **kwargs → dict
    print(args, kwargs)

total(1, 2, 3, tax=0.08)        # (1, 2, 3) {'tax': 0.08}

square = lambda x: x * x        # anonymous one-line function
square(5)                        # 25

# List / dict comprehensions
squares = [x * x for x in range(10)]
evens = [x for x in range(20) if x % 2 == 0]
lookup = {x: x * x for x in range(5)}

Error handling

try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"Error: {e}")
except (TypeError, ValueError) as e:
    print(f"Bad input: {e}")
else:
    print("No error occurred")     # runs only if no exception
finally:
    print("Always runs")            # cleanup, runs no matter what

raise ValueError("age must be positive")   # raise your own error

Object-oriented programming

class Animal:
    def __init__(self, name, sound):   # constructor
        self.name = name                # instance attribute
        self.sound = sound

    def speak(self):
        return f"{self.name} says {self.sound}"

    def __repr__(self):                 # controls how print() shows the object
        return f"Animal({self.name!r})"

class Dog(Animal):                      # inheritance
    def __init__(self, name):
        super().__init__(name, sound="Woof")

    def speak(self):                    # method override
        return f"{super().speak()}!"

d = Dog("Rex")
print(d.speak())    # "Rex says Woof!"
print(d)             # Animal('Rex')
